017package org.microbean.jersey.netty;
018
019import java.net.InetSocketAddress;
020
021import io.netty.bootstrap.ServerBootstrap;
022
023import io.netty.channel.ChannelFuture;
024import io.netty.channel.EventLoopGroup;
025
026import io.netty.channel.nio.NioEventLoopGroup;
027
028import io.netty.channel.socket.nio.NioServerSocketChannel;
029
030import io.netty.util.concurrent.Future;
031
032import org.glassfish.jersey.server.ApplicationHandler;
033
034import org.junit.Test;
035
036import static org.junit.Assume.assumeTrue;
037
038public class TestSpike {
039
040  public TestSpike() {
041    super();
042  }
043
044  @Test
045  public void testSpike() throws Exception {
046    assumeTrue(Boolean.getBoolean("runBlockingTests"));
047    final EventLoopGroup group = new NioEventLoopGroup();
048    try {
049      final ServerBootstrap serverBootstrap = new ServerBootstrap()
050        .group(group)
051        .channel(NioServerSocketChannel.class)
052        .localAddress(new InetSocketAddress("localhost", 8080))
053        .childHandler(new JerseyChannelInitializer(null, null, true, 20000000L, null, new ApplicationHandler(new Application()), 8192, null));
054      final ChannelFuture bindFuture = serverBootstrap.bind();
055      bindFuture.channel().closeFuture().addListener(c -> System.out.println("*** server closed"));
056      bindFuture.sync();
057      System.out.println("*** server started");
058      Thread.sleep(20L * 60L * 1000L); // milliseconds
059    } finally {
060      group.shutdownGracefully().addListener(f -> System.out.println("*** eventLoopGroup shutdown")).sync();
061    }
062  }
063  
064}
