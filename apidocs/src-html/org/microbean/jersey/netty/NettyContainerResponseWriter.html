148  @Override
149  public OutputStream writeResponseStatusAndHeaders(final long contentLength, final ContainerResponse containerResponse) throws ContainerException {
150    Objects.requireNonNull(containerResponse);
151    assert !this.inEventLoop();
152
153    this.contentLength = contentLength;
154
155    final OutputStream returnValue;
156
157    // Jersey's native Netty support does this.  I don't know why.  We
158    // follow suit; there's probably an edge case that gave rise to
159    // this.
160    final boolean responseWritten = this.responseWritten;
161    if (responseWritten) {
162      returnValue = null;
163    } else {
164      this.responseWritten = true;
165
166      final String reasonPhrase = containerResponse.getStatusInfo().getReasonPhrase();
167      final int statusCode = containerResponse.getStatus();
168      final HttpResponseStatus status = reasonPhrase == null ? HttpResponseStatus.valueOf(statusCode) : new HttpResponseStatus(statusCode, reasonPhrase);
169
170      final HttpResponse httpResponse;
171      if (contentLength == 0L) {
172        httpResponse = new DefaultFullHttpResponse(this.httpRequest.protocolVersion(), status);
173      } else {
174        httpResponse = new DefaultHttpResponse(this.httpRequest.protocolVersion(), status);
175      }
176      
177      if (contentLength < 0L) {
178        HttpUtil.setTransferEncodingChunked(httpResponse, true);
179      } else {
180        HttpUtil.setContentLength(httpResponse, contentLength);
181      }
182
183      final Map<? extends String, ? extends List<String>> containerResponseHeaders = containerResponse.getStringHeaders();
184      if (containerResponseHeaders != null && !containerResponseHeaders.isEmpty()) {
185        final Collection<? extends Entry<? extends String, ? extends List<String>>> entrySet = containerResponseHeaders.entrySet();
186        if (entrySet != null && !entrySet.isEmpty()) {
187          final HttpHeaders headers = httpResponse.headers();
188          for (final Entry<? extends String, ? extends List<String>> entry : entrySet) {
189            if (entry != null) {
190              headers.add(entry.getKey(), entry.getValue());
191            }
192          }
193        }
194      }
195
196      if (HttpUtil.isKeepAlive(this.httpRequest)) {
197        HttpUtil.setKeepAlive(httpResponse, true);
198      }
199
200      // Enqueue a task to write and flush the headers on the event
201      // loop.
202      channelHandlerContext.writeAndFlush(httpResponse);
203
204      if (this.needsOutputStream(contentLength)) {
205
206        // We've determined that there is a payload/entity.  Our
207        // ultimate responsibility is to return an OutputStream the
208        // end user's Jersey resource class will write to.
209        //
210        // Allocate a ByteBuf suitable for doing IO.  This could be
211        // heap-based or native.  We don't care; we trust Netty.
212        final ByteBuf byteBuf;
213        if (contentLength > 0L && contentLength <= Integer.MAX_VALUE) {
214          byteBuf = this.channelHandlerContext.alloc().ioBuffer((int)contentLength);
215        } else {
216          assert contentLength < 0L;
217          byteBuf = this.channelHandlerContext.alloc().ioBuffer();
218        }
219        assert byteBuf != null;
220
221        // Ensure that this buffer is released when/if the channel is
222        // closed.
223        channelHandlerContext.channel().closeFuture().addListener(ignored -> byteBuf.release());
224
225        // A ChunkedInput despite its name has nothing to do with
226        // chunked Transfer-Encoding.  It is usable anywhere you like.
227        // So here, because writes are happening on an OutputStream in
228        // Jersey-land on a separate thread, and are then being run on
229        // the event loop, unless we were to block the event loop and
230        // wait for the OutputStream to close, we wouldn't know when
231        // to do the write.  So we use ChunkedInput even in cases
232        // where the Content-Length is set to a positive integer.
233        final ChunkedInput<?> chunkedInput = this.createChunkedInput(this.channelHandlerContext.executor(), byteBuf, contentLength);
234        if (chunkedInput == null) {
235          throw new IllegalStateException("createChunkedInput() == null");
236        }
237        this.chunkedInput = chunkedInput;
238
239        // Enqueue a task that will query the ByteBufChunkedInput for
240        // its chunks via its readChunk() method on the event loop.
241        channelHandlerContext.write(this.chunkedInput);
242
243        // Then return an OutputStream implementation that writes to
244        // the very same ByteBuf, ensuring that writes take place on
245        // the event loop.  The net effect is that as this stream
246        // writes to the ByteBuf, the ChunkedWriteHandler consuming
247        // the ChunkedInput by way of its readChunk() method, also on
248        // the event loop, will stream the results as they are made
249        // available.
250        //
251        // TODO: replace the null third parameter with a
252        // GenericFutureListener that will Do The Right Thing™ with
253        // any exceptions.  **Remember that this listener will be
254        // invoked on the event loop.**
255        returnValue = new EventLoopPinnedByteBufOutputStream(this.channelHandlerContext.executor(), byteBuf, null);
256      } else {
257        returnValue = null;
258      }
259    }
260    return returnValue;
261  }

263  /**
264   * Creates a new {@link ChunkedInput} instance whose {@link
265   * ChunkedInput#readChunk(ByteBufAllocator)} method will stream
266   * {@link ByteBuf} chunks to the Netty transport.
267   *
268   * <p>This method never returns {@code null}.</p>
269   *
270   * <p>In normal usage this method will be invoked on a thread that
271   * is <strong>not</strong> {@linkplain EventExecutor#inEventLoop()
272   * in the Netty event loop}.  Care <strong>must</strong> be taken if
273   * an override of this method decides to invoke any methods on the
274   * supplied {@link ByteBuf} to ensure those methods are invoked in
275   * the Netty event loop.  This implementation does not invoke any
276   * {@link ByteBuf} methods and overrides are strongly urged to
277   * follow suit.</p>
278   *
279   * <p>Overrides of this method must not return {@code null}.</p>
280   *
281   * <p>This method is called from the {@link
282   * #writeResponseStatusAndHeaders(long, ContainerResponse)} method.
283   * Overrides must not call that method or an infinite loop may
284   * result.</p>
285   *
286   * @param eventExecutor an {@link EventExecutor}, supplied for
287   * convenience, that can be used to ensure certain tasks are
288   * executed in the Netty event loop; must not be {@code null}
289   *
290   * @param source the {@link ByteBuf}, <strong>which might be
291   * mutating</strong> in the Netty event loop, that the returned
292   * {@link ChunkedInput} should read from in some way when its {@link
293   * ChunkedInput#readChunk(ByteBufAllocator)} method is called from
294   * the Netty event loop; must not be {@code null}
295   *
296   * @param contentLength a {@code long} representing the value of any
297   * {@code Content-Length} header that might have been present; it is
298   * guaranteed that when this method is invoked by the default
299   * implementation of the {@link #writeResponseStatusAndHeaders(long,
300   * ContainerResponse)} method this parameter will never be {@code
301   * 0L} but might very well be less than {@code 0L} to indicate an
302   * unknown content length
303   *
304   * @return a new {@link ChunkedInput} that reads in some way from
305   * the supplied {@code source} when its {@link
306   * ChunkedInput#readChunk(ByteBufAllocator)} method is called from
307   * the Netty event loop; never {@code null}
308   *
309   * @see ByteBufChunkedInput#ByteBufChunkedInput(ByteBuf, long)
310   *
311   * @see ChunkedInput#readChunk(ByteBufAllocator)
312   */
313  protected ChunkedInput<?> createChunkedInput(final EventExecutor eventExecutor, final ByteBuf source, final long contentLength) {
314    return new ByteBufChunkedInput(source, contentLength);
315  }

317  @Override
318  public final boolean suspend(final long timeout, final TimeUnit timeUnit, final TimeoutHandler timeoutHandler) {
319    assert !this.inEventLoop();
320
321    // Lifted from Jersey's supplied Netty integration, with repairs.
322    final boolean returnValue;
323    if (this.suspendTimeoutHandler != null) {
324      returnValue = false;
325    } else {
326      this.suspendTimeoutHandler = () -> {
327        timeoutHandler.onTimeout(this);
328        // TODO: not sure about this
329        this.suspendTimeoutHandler = null;
330      };
331      if (timeout > 0) {
332        this.suspendTimeoutFuture =
333          this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
334      }
335      returnValue = true;
336    }
337    return returnValue;
338  }
339
340  @Override
341  public final void setSuspendTimeout(final long timeout, final TimeUnit timeUnit) {
342    assert !this.inEventLoop();
343
344    // Lifted from Jersey's supplied Netty integration, with repairs.
345    if (this.suspendTimeoutHandler == null) {
346      throw new IllegalStateException();
347    }
348    if (this.suspendTimeoutFuture != null) {
349      this.suspendTimeoutFuture.cancel(true);
350    }
351    if (timeout > 0) {
352      this.suspendTimeoutFuture =
353        this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
354    }
355  }

357  /**
358   * Invoked on a non-event-loop thread by Jersey when, as far as
359   * Jersey is concerned, all processing has completed successfully.
360   *
361   * <p>This implementation ensures that {@link ChunkedInput#close()}
362   * is called on the Netty event loop on the {@link ChunkedInput}
363   * returned by the {@link #createChunkedInput(EventExecutor,
364   * ByteBuf, long)} method, and that a {@link
365   * LastHttpContent#EMPTY_LAST_CONTENT} message is written
366   * immediately afterwards, and that the {@link
367   * ChannelHandlerContext} is {@linkplain
368   * ChannelHandlerContext#flush() flushed}.  All of these invocations
369   * occur on the event loop.</p>
370   *
371   * @see ChunkedInput#close()
372   *
373   * @see #createChunkedInput(EventExecutor, ByteBuf, long)
374   *
375   * @see LastHttpContent#EMPTY_LAST_CONTENT
376   *
377   * @see ChannelHandlerContext#write(Object)
378   *
379   * @see ChannelHandlerContext#flush()
380   *
381   * @see ContainerResponseWriter#commit()
382   */
383  @Override
384  public final void commit() {
385    assert !this.inEventLoop();
386    final ChunkedInput<?> chunkedInput = this.chunkedInput;
387    this.chunkedInput = null;
388    this.channelHandlerContext.executor().submit((Callable<Void>)() -> {
389        try {
390          if (chunkedInput != null) {
391            chunkedInput.close();
392            channelHandlerContext.write(LastHttpContent.EMPTY_LAST_CONTENT);
393          }
394        } finally {
395          this.channelHandlerContext.flush();
396        }
397        return null;
398      });
399  }

401  /**
402   * Invoked on a non-event-loop thread by Jersey when, as far as
403   * Jersey is concerned, all processing has completed unsuccessfully.
404   *
405   * @param throwable the {@link Throwable} that caused failure;
406   * strictly speaking may be {@code null}
407   *
408   * @exception RuntimeException if an error occurs while processing
409   * the supplied {@link Throwable}; the supplied {@link Throwable}
410   * will be {@linkplain Throwable#addSuppressed(Throwable) added as a
411   * suppressed <code>Throwable</code>}
412   *
413   * @exception Error if an error occurs while processing the supplied
414   * {@link Throwable}; the supplied {@link Throwable} will be
415   * {@linkplain Throwable#addSuppressed(Throwable) added as a
416   * suppressed <code>Throwable</code>}
417   *
418   * @see ContainerResponseWriter#failure(Throwable)
419   */
420  @Override
421  public void failure(final Throwable throwable) {
422    assert !this.inEventLoop();
423
424    try {
425      final HttpMessage httpMessage = new DefaultFullHttpResponse(this.httpRequest.protocolVersion(),
426                                                              HttpResponseStatus.INTERNAL_SERVER_ERROR);
427      HttpUtil.setContentLength(httpMessage, 0L);
428      this.channelHandlerContext.writeAndFlush(httpMessage).addListener(ChannelFutureListener.CLOSE);
429    } catch (final RuntimeException | Error throwMe) {
430      if (throwable != null) {
431        throwMe.addSuppressed(throwable);
432      }
433      throw throwMe;
434    }
435  }

469  private final boolean needsOutputStream(final long contentLength) {
470    return contentLength != 0 && !HttpMethod.HEAD.equals(this.methodSupplier.get());
471  }
472
473  private final boolean inEventLoop() {
474    return this.channelHandlerContext.executor().inEventLoop();
