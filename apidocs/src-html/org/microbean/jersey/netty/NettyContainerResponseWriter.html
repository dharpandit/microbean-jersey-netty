149  @Override
150  public OutputStream writeResponseStatusAndHeaders(final long contentLength, final ContainerResponse containerResponse) throws ContainerException {
151    Objects.requireNonNull(containerResponse);
152    assert !this.inEventLoop();
153
154    final OutputStream returnValue;
155
156    // Jersey's native Netty support does this.  I don't know why.  We
157    // follow suit; there's probably an edge case that gave rise to
158    // this.
159    final boolean responseWritten = this.responseWritten;
160    if (responseWritten) {
161      returnValue = null;
162    } else {
163      this.responseWritten = true;
164
165      final String reasonPhrase = containerResponse.getStatusInfo().getReasonPhrase();
166      final int statusCode = containerResponse.getStatus();
167      final HttpResponseStatus status = reasonPhrase == null ? HttpResponseStatus.valueOf(statusCode) : new HttpResponseStatus(statusCode, reasonPhrase);
168
169      final HttpResponse httpResponse;
170      if (contentLength == 0L) {
171        httpResponse = new DefaultFullHttpResponse(this.httpRequest.protocolVersion(), status);
172      } else {
173        httpResponse = new DefaultHttpResponse(this.httpRequest.protocolVersion(), status);
174      }
175      
176      if (contentLength < 0L) {
177        HttpUtil.setTransferEncodingChunked(httpResponse, true);
178      } else {
179        HttpUtil.setContentLength(httpResponse, contentLength);
180      }
181
182      final Map<? extends String, ? extends List<String>> containerResponseHeaders = containerResponse.getStringHeaders();
183      if (containerResponseHeaders != null && !containerResponseHeaders.isEmpty()) {
184        final Collection<? extends Entry<? extends String, ? extends List<String>>> entrySet = containerResponseHeaders.entrySet();
185        if (entrySet != null && !entrySet.isEmpty()) {
186          final HttpHeaders headers = httpResponse.headers();
187          for (final Entry<? extends String, ? extends List<String>> entry : entrySet) {
188            if (entry != null) {
189              headers.add(entry.getKey(), entry.getValue());
190            }
191          }
192        }
193      }
194
195      if (HttpUtil.isKeepAlive(this.httpRequest)) {
196        HttpUtil.setKeepAlive(httpResponse, true);
197      }
198
199      // Enqueue a task to write and flush the headers on the event
200      // loop.
201      //
202      // TODO: suppose this flush goes out with a 200.  And then the
203      // output stream writes a bunch of crap, or blows up.  We want
204      // to "retract" this write but it's too late.  On the other hand
205      // we don't want to buffer everything I don't think.
206      channelHandlerContext.writeAndFlush(httpResponse);
207
208      if (this.needsOutputStream(contentLength)) {
209
210        // We've determined that there is a payload/entity.  Our
211        // ultimate responsibility is to return an OutputStream the
212        // end user's Jersey resource class will write to.
213        //
214        // Allocate a ByteBuf suitable for doing IO.  This could be
215        // heap-based or native.  We don't care; we trust Netty.
216        final ByteBuf byteBuf;
217        if (contentLength > 0L && contentLength <= Integer.MAX_VALUE) {
218          // Positive content length.
219          byteBuf = this.channelHandlerContext.alloc().ioBuffer((int)contentLength);
220        } else {
221          // Negative content length or ridiculously huge content
222          // length so ignore it for capacity purposes.
223          assert contentLength != 0L;
224          byteBuf = this.channelHandlerContext.alloc().ioBuffer();
225        }
226        assert byteBuf != null;
227
228        // We store a reference to it ONLY so that we can release it
229        // at a later date.
230        this.byteBuf = byteBuf;
231
232        // A ChunkedInput despite its name has nothing to do with
233        // chunked Transfer-Encoding.  It is usable anywhere you like.
234        // So here, because writes are happening on an OutputStream in
235        // Jersey-land on a separate thread, and are then being run on
236        // the event loop, unless we were to block the event loop and
237        // wait for the OutputStream to close, we wouldn't know when
238        // to do the write.  So we use ChunkedInput even in cases
239        // where the Content-Length is set to a positive integer.
240        final ChunkedInput<?> chunkedInput = this.createChunkedInput(this.channelHandlerContext.executor(), byteBuf, contentLength);
241        if (chunkedInput == null) {
242          this.byteBuf = null;
243          byteBuf.release();
244          throw new IllegalStateException("createChunkedInput() == null");
245        }
246        this.chunkedInput = chunkedInput;
247
248        // Enqueue a task that will query the ByteBufChunkedInput for
249        // its chunks via its readChunk() method on the event loop.
250        channelHandlerContext.write(this.chunkedInput);
251
252        // Then return an OutputStream implementation that writes to
253        // the very same ByteBuf, ensuring that writes take place on
254        // the event loop.  The net effect is that as this stream
255        // writes to the ByteBuf, the ChunkedWriteHandler consuming
256        // the ChunkedInput by way of its readChunk() method, also on
257        // the event loop, will stream the results as they are made
258        // available.
259        //
260        // TODO: replace the null third parameter with a
261        // GenericFutureListener that will Do The Right Thing™ with
262        // any exceptions.  **Remember that this listener will be
263        // invoked on the event loop.**
264        returnValue =
265          new EventLoopPinnedByteBufOutputStream(this.channelHandlerContext.executor(),
266                                                 byteBuf,
267                                                 null);
268      } else {
269        returnValue = null;
270      }
271    }
272    return returnValue;
273  }

275  /**
276   * Creates and returns a new {@link ChunkedInput} instance whose
277   * {@link ChunkedInput#readChunk(ByteBufAllocator)} method will
278   * stream {@link ByteBuf} chunks to the Netty transport.
279   *
280   * <p>This method never returns {@code null}.</p>
281   *
282   * <p>In normal usage this method will be invoked on a thread that
283   * is <strong>not</strong> {@linkplain EventExecutor#inEventLoop()
284   * in the Netty event loop}.  Care <strong>must</strong> be taken if
285   * an override of this method decides to invoke any methods on the
286   * supplied {@link ByteBuf} to ensure those methods are invoked in
287   * the Netty event loop.  This implementation does not invoke any
288   * {@link ByteBuf} methods and overrides are strongly urged to
289   * follow suit.</p>
290   *
291   * <p>Overrides of this method must not return {@code null}.</p>
292   *
293   * <p>This method is called from the {@link
294   * #writeResponseStatusAndHeaders(long, ContainerResponse)} method.
295   * Overrides must not call that method or an infinite loop may
296   * result.</p>
297   *
298   * @param eventExecutor an {@link EventExecutor}, supplied for
299   * convenience, that can be used to ensure certain tasks are
300   * executed in the Netty event loop; must not be {@code null}
301   *
302   * @param source the {@link ByteBuf}, <strong>which might be
303   * mutating</strong> in the Netty event loop, that the returned
304   * {@link ChunkedInput} should read from in some way when its {@link
305   * ChunkedInput#readChunk(ByteBufAllocator)} method is called from
306   * the Netty event loop; must not be {@code null}
307   *
308   * @param contentLength a {@code long} representing the value of any
309   * {@code Content-Length} header that might have been present; it is
310   * guaranteed that when this method is invoked by the default
311   * implementation of the {@link #writeResponseStatusAndHeaders(long,
312   * ContainerResponse)} method this parameter will never be {@code
313   * 0L} but might very well be less than {@code 0L} to indicate an
314   * unknown content length
315   *
316   * @return a new {@link ChunkedInput} that reads in some way from
317   * the supplied {@code source} when its {@link
318   * ChunkedInput#readChunk(ByteBufAllocator)} method is called from
319   * the Netty event loop; never {@code null}
320   *
321   * @see ByteBufChunkedInput#ByteBufChunkedInput(ByteBuf, long)
322   *
323   * @see ChunkedInput#readChunk(ByteBufAllocator)
324   */
325  protected ChunkedInput<ByteBuf> createChunkedInput(final EventExecutor eventExecutor, final ByteBuf source, final long contentLength) {
326    return new ByteBufChunkedInput(source, contentLength);
327  }

329  @Override
330  public final boolean suspend(final long timeout, final TimeUnit timeUnit, final TimeoutHandler timeoutHandler) {
331    assert !this.inEventLoop();
332
333    // Lifted from Jersey's supplied Netty integration, with repairs.
334    final boolean returnValue;
335    if (this.suspendTimeoutHandler != null) {
336      returnValue = false;
337    } else {
338      this.suspendTimeoutHandler = () -> {
339        timeoutHandler.onTimeout(this);
340        // TODO: not sure about this
341        this.suspendTimeoutHandler = null;
342      };
343      if (timeout > 0) {
344        this.suspendTimeoutFuture =
345          this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
346      }
347      returnValue = true;
348    }
349    return returnValue;
350  }
351
352  @Override
353  public final void setSuspendTimeout(final long timeout, final TimeUnit timeUnit) {
354    assert !this.inEventLoop();
355
356    // Lifted from Jersey's supplied Netty integration, with repairs.
357    if (this.suspendTimeoutHandler == null) {
358      throw new IllegalStateException();
359    }
360    if (this.suspendTimeoutFuture != null) {
361      this.suspendTimeoutFuture.cancel(true);
362    }
363    if (timeout > 0) {
364      this.suspendTimeoutFuture =
365        this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
366    }
367  }

369  /**
370   * Invoked on a non-event-loop thread by Jersey when, as far as
371   * Jersey is concerned, all processing has completed successfully.
372   *
373   * <p>This implementation ensures that {@link ChunkedInput#close()}
374   * is called on the Netty event loop on the {@link ChunkedInput}
375   * returned by the {@link #createChunkedInput(EventExecutor,
376   * ByteBuf, long)} method, and that a {@link
377   * LastHttpContent#EMPTY_LAST_CONTENT} message is written
378   * immediately afterwards, and that the {@link
379   * ChannelHandlerContext} is {@linkplain
380   * ChannelHandlerContext#flush() flushed}.  All of these invocations
381   * occur on the event loop.</p>
382   *
383   * @see ChunkedInput#close()
384   *
385   * @see #createChunkedInput(EventExecutor, ByteBuf, long)
386   *
387   * @see LastHttpContent#EMPTY_LAST_CONTENT
388   *
389   * @see ChannelHandlerContext#write(Object)
390   *
391   * @see ChannelHandlerContext#flush()
392   *
393   * @see ContainerResponseWriter#commit()
394   */
395  @Override
396  public final void commit() {
397    assert !this.inEventLoop();
398    final ChunkedInput<?> chunkedInput = this.chunkedInput;
399    this.chunkedInput = null;
400    final ReferenceCounted byteBuf = this.byteBuf;
401    this.byteBuf = null;
402    this.channelHandlerContext.executor().submit((Callable<Void>)() -> {
403        assert inEventLoop();
404        if (chunkedInput == null) {
405          assert byteBuf == null;
406        } else {
407          assert byteBuf != null;
408          assert byteBuf.refCnt() == 1;
409          
410          // Mark our ChunkedInput as *closed to new input*.  It may
411          // still contain contents that need to be read.  (This
412          // effectively flips an internal switch in the ChunkedInput
413          // that allows for isEndOfInput() to one day return true.)
414          chunkedInput.close();
415          
416          // This will tell the ChunkedWriteHandler outbound from us
417          // to "drain" our ByteBufChunkedInput via successive calls
418          // to #readChunk(ByteBufAllocator).  The isEndOfInput()
419          // method will be called as part of this process and will
420          // eventually return true.
421          channelHandlerContext.flush();
422          
423          // Assert that the ChunkedInput is drained and release its
424          // wrapped ByteBuf.
425          assert chunkedInput.isEndOfInput();
426          final boolean released = byteBuf.release();
427          assert released;
428          
429          // Send the magic message that tells the HTTP machinery to
430          // finish up.
431          channelHandlerContext.writeAndFlush(LastHttpContent.EMPTY_LAST_CONTENT);
432        }
433        return null;
434      });
435  }

437  /**
438   * Invoked on a non-event-loop thread by Jersey when, as far as
439   * Jersey is concerned, all processing has completed unsuccessfully.
440   *
441   * @param throwable the {@link Throwable} that caused failure;
442   * strictly speaking may be {@code null}
443   *
444   * @exception RuntimeException if an error occurs while processing
445   * the supplied {@link Throwable}; the supplied {@link Throwable}
446   * will be {@linkplain Throwable#addSuppressed(Throwable) added as a
447   * suppressed <code>Throwable</code>}
448   *
449   * @exception Error if an error occurs while processing the supplied
450   * {@link Throwable}; the supplied {@link Throwable} will be
451   * {@linkplain Throwable#addSuppressed(Throwable) added as a
452   * suppressed <code>Throwable</code>}
453   *
454   * @see ContainerResponseWriter#failure(Throwable)
455   */
456  @Override
457  public void failure(final Throwable throwable) {
458    assert !this.inEventLoop();
459
460    try {
461      final HttpMessage httpMessage = new DefaultFullHttpResponse(this.httpRequest.protocolVersion(),
462                                                                  HttpResponseStatus.INTERNAL_SERVER_ERROR);
463      HttpUtil.setContentLength(httpMessage, 0L);
464      final ChunkedInput<?> chunkedInput = this.chunkedInput;
465      this.chunkedInput = null;
466      final ReferenceCounted byteBuf = this.byteBuf;
467      this.byteBuf = null;
468      this.channelHandlerContext.executor().submit(() -> {
469          try {
470            assert inEventLoop();
471            channelHandlerContext.write(httpMessage);
472            if (chunkedInput == null) {
473              assert byteBuf == null;
474            } else {
475              assert byteBuf != null;
476              assert byteBuf.refCnt() == 1;
477              final boolean released = byteBuf.release();
478              assert released;
479            }
480          } catch (final RuntimeException | Error throwMe) {
481            if (throwable != null) {
482              throwMe.addSuppressed(throwable);
483            }
484            throw throwMe;
485          } finally {
486            channelHandlerContext.flush();
487            channelHandlerContext.close();
488          }
489          return null;
490        });
491    } catch (final RuntimeException | Error throwMe) {
492      if (throwable != null) {
493        throwMe.addSuppressed(throwable);
494      }
495      throw throwMe;
496    }
497  }

531  private final boolean needsOutputStream(final long contentLength) {
532    return contentLength != 0 && !HttpMethod.HEAD.equals(this.methodSupplier.get());
533  }
534
535  private final boolean inEventLoop() {
536    return this.channelHandlerContext.executor().inEventLoop();
