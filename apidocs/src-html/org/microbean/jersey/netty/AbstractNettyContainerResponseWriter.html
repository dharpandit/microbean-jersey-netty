171  /**
172   * Implements the {@link
173   * ContainerResponseWriter#writeResponseStatusAndHeaders(long,
174   * ContainerResponse)} method by calling the {@link
175   * #writeAndFlushStatusAndHeaders(ContainerResponse, long,
176   * ChannelPromise)} method, then {@linkplain
177   * #needsOutputStream(long) determining whether an
178   * <code>OutputStream</code> is needed}, and allocating and
179   * returning a {@link EventLoopPinnedByteBufOutputStream} that works
180   * properly with the Netty machinery if so.
181   *
182   * <p>This method may return {@code null}.</p>
183   *
184   * @param contentLength the length in bytes that will be written to
185   * the {@link OutputStream} that will be returned; may be {@code
186   * -1L} to indicate an unknown content length but otherwise must be
187   * zero or a positive {@code long}
188   *
189   * @param containerResponse the {@link ContainerResponse} for which
190   * a write is to take place; must not be {@code null}
191   *
192   * @return an {@link OutputStream}, or {@code null}
193   *
194   * @exception NullPointerException if {@code containerResponse} is
195   * {@code null}
196   *
197   * @exception IllegalStateException if a subclass has overridden
198   * {@link #createChunkedInput(EventExecutor, ByteBuf, long)} to
199   * return {@code null}
200   *
201   * @exception ContainerException if an error occurs
202   *
203   * @see #writeAndFlushStatusAndHeaders(ContainerResponse, long,
204   * ChannelPromise)
205   *
206   * @see #needsOutputStream(long)
207   *
208   * @see #createChunkedInput(EventExecutor, ByteBuf, long)
209   *
210   * @see EventLoopPinnedByteBufOutputStream
211   */
212  @Override
213  public final OutputStream writeResponseStatusAndHeaders(final long contentLength, final ContainerResponse containerResponse) {
214    final String mn = "writeResponseStatusAndHeaders";
215    if (logger.isLoggable(Level.FINER)) {
216      logger.entering(cn, mn, new Object[] { Long.valueOf(contentLength), containerResponse });
217    }
218
219    Objects.requireNonNull(containerResponse);
220    assert !this.inEventLoop();
221
222    final ChannelPromise statusAndHeadersPromise = this.channelHandlerContext.newPromise();
223    assert statusAndHeadersPromise != null;
224    statusAndHeadersPromise.addListener(listener);
225
226    this.writeAndFlushStatusAndHeaders(containerResponse, contentLength, statusAndHeadersPromise);
227
228    final OutputStream returnValue;
229    if (this.needsOutputStream(contentLength)) {
230      assert contentLength != 0L;
231
232      // We've determined that there is a payload/entity.  Our
233      // ultimate responsibility is to return an OutputStream the
234      // end user's Jersey resource class will write to.
235      //
236      // Allocate a ByteBuf suitable for doing IO.  This could be
237      // heap-based or native.  We don't care; we trust Netty.
238      final ByteBuf byteBuf;
239      if (contentLength > 0L && contentLength <= Integer.MAX_VALUE) {
240        // Positive content length.
241        byteBuf = this.channelHandlerContext.alloc().ioBuffer((int)contentLength);
242      } else {
243        // Negative content length or ridiculously huge content
244        // length so ignore it for capacity purposes.
245        byteBuf = this.channelHandlerContext.alloc().ioBuffer();
246      }
247      assert byteBuf != null;
248
249      // A ChunkedInput despite its name has nothing to do with
250      // chunked Transfer-Encoding.  It is usable anywhere you like.
251      // So here, because writes are happening on an OutputStream in
252      // Jersey-land on a separate thread, and are then being run on
253      // the event loop, unless we were to block the event loop and
254      // wait for the OutputStream to close, we wouldn't know when
255      // to do the write.  So we use ChunkedInput even in cases
256      // where the Content-Length is set to a positive integer.
257      final BoundedChunkedInput<?> chunkedInput = this.createChunkedInput(this.channelHandlerContext.executor(), byteBuf, contentLength);
258      if (chunkedInput == null) {
259        // A user's implementation of createChunkedInput() behaved
260        // badly.  Clean up and bail out.
261        byteBuf.release();
262        throw new IllegalStateException("createChunkedInput() == null");
263      }
264      this.chunkedInput = chunkedInput;
265
266      // We store a reference to the allocated ByteBuf ONLY so that we
267      // can release it at #failure(Throwable) time or if
268      // createChunkedInput() returns null incorrectly.  Otherwise the
269      // BoundedChunkedInput's close() method is responsible for
270      // releasing it properly.
271      this.byteBuf = byteBuf;
272
273      // Enqueue a task that will query the ChunkedInput for
274      // its chunks via its readChunk() method on the event loop.
275      final ChannelPromise chunkedInputWritePromise = this.channelHandlerContext.newPromise();
276      assert chunkedInputWritePromise != null;
277      chunkedInputWritePromise.addListener(listener);
278      channelHandlerContext.write(chunkedInput, chunkedInputWritePromise);
279
280      // Then return an OutputStream implementation that writes to the
281      // very same ByteBuf.  The net effect is that as this stream
282      // writes to the ByteBuf, the ChunkedWriteHandler consuming the
283      // ChunkedInput by way of its readChunk() method, on the event
284      // loop, will stream the results as they are made available.  We
285      // can get away with this
286      // two-threads-interacting-with-the-same-ByteBuf-without-synchronization
287      // use case because one thread is affecting only the ByteBuf's
288      // reader index, and the other is affecting only the ByteBuf's
289      // writer index.  See
290      // https://stackoverflow.com/questions/58403725/although-bytebuf-is-not-guaranteed-to-be-thread-safe-will-this-use-case-be-ok
291      // for more.
292      returnValue = new ByteBufOutputStream(byteBuf);
293    } else {
294      returnValue = null;
295    }
296
297    if (logger.isLoggable(Level.FINER)) {
298      logger.exiting(cn, mn, returnValue);
299    }
300    return returnValue;
301  }

303  /**
304   * Invoked on a non-event-loop thread by Jersey when, as far as
305   * Jersey is concerned, all processing has completed successfully.
306   *
307   * <p>This implementation ensures that {@link
308   * BoundedChunkedInput#setEndOfInput()} is called on the Netty event
309   * loop on the {@link BoundedChunkedInput} returned by the {@link
310   * #createChunkedInput(EventExecutor, ByteBuf, long)} method, and
311   * that a {@linkplain #writeLastContentMessage(ChannelPromise) final
312   * content message is written immediately afterwards}, and that the
313   * {@link ChannelHandlerContext} is {@linkplain
314   * ChannelHandlerContext#flush() flushed}.  This will result in a
315   * downstream {@link ChunkedWriteHandler} writing the underlying
316   * data and calling {@link ChunkedInput#close()} at the proper time.
317   * All of these invocations occur on the event loop.</p>
318   *
319   * @see ChunkedInput#close()
320   *
321   * @see #createChunkedInput(EventExecutor, ByteBuf, long)
322   *
323   * @see #writeLastContentMessage(ChannelPromise)
324   *
325   * @see ChannelHandlerContext#write(Object)
326   *
327   * @see ChannelHandlerContext#flush()
328   *
329   * @see ContainerResponseWriter#commit()
330   *
331   * @see ChunkedWriteHandler
332   */
333  @Override
334  public final void commit() {
335    final String mn = "commit";
336    if (logger.isLoggable(Level.FINER)) {
337      logger.entering(cn, mn);
338    }
339
340    assert !this.inEventLoop();
341    final BoundedChunkedInput<?> chunkedInput = this.chunkedInput;
342    this.chunkedInput = null;
343    if (chunkedInput != null) {
344      final ChannelPromise promise = this.channelHandlerContext.newPromise();
345      assert promise != null;
346      promise.addListener(listener);
347      this.channelHandlerContext.executor().submit((Callable<Void>)() -> {
348          assert inEventLoop();
349          chunkedInput.setEndOfInput();
350          this.writeLastContentMessage(promise);
351          channelHandlerContext.flush();
352          return null;
353        }).addListener(listener);
354    }
355
356    if (logger.isLoggable(Level.FINER)) {
357      logger.exiting(cn, mn);
358    }
359  }

361  /**
362   * Invoked on a non-event-loop thread by Jersey when, as far as
363   * Jersey is concerned, all processing has completed unsuccessfully.
364   *
365   * <p><strong>This method never returns.  It always throws a {@link
366   * RuntimeException} or an {@link Error} of some
367   * variety.</strong></p>
368   *
369   * <h2>Design Notes</h2>
370   *
371   * <p><a
372   * href="https://github.com/jax-rs/spec/blob/b4b0d91fc902d58dfa1670b2ee1bc34a640bb6f5/chapters/resources.tex#L143-L144"
373   * target="_parent">JAX-RS: Java&trade; API for RESTful Web
374   * Services, section 3.3.4</a> governs this method, but it is
375   * somewhat ambiguous whether this method represents the container
376   * itself.  All Jersey-supplied {@link ContainerResponseWriter}
377   * implementations (except their <a
378   * href="https://github.com/eclipse-ee4j/jersey/blob/eafb9bdcb82dfa3fd76dd957d307b99d4a22c87f/containers/netty-http/src/main/java/org/glassfish/jersey/netty/httpserver/NettyResponseWriter.java#L177-L181"
379   * target="_parent">Netty container</a>, which seems to be <a
380   * href="https://github.com/jersey/jersey/pull/3791"
381   * target="_parent">buggy in this regard</a>) rethrow the supplied
382   * {@link Throwable} as some kind of {@link RuntimeException} from
383   * their implementations of this method, so we follow suit.</p>
384   *
385   * @param throwable the {@link Throwable} that caused failure;
386   * strictly speaking may be {@code null}
387   *
388   * @exception ContainerException <strong>thrown whenever this method
389   * completes without further problems, or if the supplied {@link
390   * Throwable} is itself a {@link ContainerException}, in order to
391   * comply with section 3.3.4 of the JAX-RS 2.1
392   * specification</strong>
393   *
394   * @exception RuntimeException if the supplied {@link Throwable} is
395   * itself a {@link RuntimeException} and this method completes
396   * without further problems, or if an error occurs while processing
397   * the supplied {@link Throwable} in which case the supplied {@link
398   * Throwable} will be {@linkplain Throwable#addSuppressed(Throwable)
399   * added as a suppressed <code>Throwable</code>}
400   *
401   * @exception Error if the supplied {@link Throwable} is itself an
402   * {@link Error} and this method completes without further problems,
403   * or if an error occurs while processing the supplied {@link
404   * Throwable} in which case the supplied {@link Throwable} will be
405   * {@linkplain Throwable#addSuppressed(Throwable) added as a
406   * suppressed <code>Throwable</code>}
407   *
408   * @see ContainerResponseWriter#failure(Throwable)
409   *
410   * @see <a
411   * href="https://github.com/jax-rs/spec/blob/b4b0d91fc902d58dfa1670b2ee1bc34a640bb6f5/chapters/resources.tex#L143-L144"
412   * target="_parent">JAX-RS: Java&trade; API for RESTful Web
413   * Services, section 3.3.4</a>
414   */
415  @Override
416  public final void failure(final Throwable throwable) {
417    final String mn = "failure";
418    if (logger.isLoggable(Level.FINER)) {
419      logger.entering(cn, mn, throwable);
420    }
421
422    assert !this.inEventLoop();
423
424    this.chunkedInput = null;
425    final ReferenceCounted byteBuf = this.byteBuf;
426    this.byteBuf = null;
427    try {
428      final ChannelPromise promise = this.channelHandlerContext.newPromise();
429      assert promise != null;
430      promise.addListener(listener);
431      this.channelHandlerContext.executor().submit((Callable<Void>)() -> {
432          try {
433            assert inEventLoop();
434            this.writeFailureMessage(promise);
435          } catch (final RuntimeException | Error throwMe) {
436            if (throwable != null) {
437              throwMe.addSuppressed(throwable);
438            }
439            throw throwMe;
440          } finally {
441            try {
442              try {
443                channelHandlerContext.flush();
444              } catch (final RuntimeException | Error throwMe2) {
445                if (throwable != null) {
446                  throwMe2.addSuppressed(throwable);
447                }
448                throw throwMe2;
449              }
450              try {
451                channelHandlerContext.close();
452              } catch (final RuntimeException | Error throwMe2) {
453                if (throwable != null) {
454                  throwMe2.addSuppressed(throwable);
455                }
456                throw throwMe2;
457              }
458            } finally {
459              assert byteBuf != null;
460              assert byteBuf.refCnt() == 1;
461              final boolean released = byteBuf.release();
462              assert released;
463            }
464          }
465          return null;
466        }).addListener(listener);
467    } catch (final RuntimeException | Error throwMe) {
468      // There was a problem submitting the task to the Netty
469      // infrastructure.  Make sure we don't lose the original
470      // Throwable.
471      if (throwable != null) {
472        throwMe.addSuppressed(throwable);
473      }
474      throw throwMe;
475    }
476
477    // See
478    // https://github.com/eclipse-ee4j/jersey/blob/b7fbb3e75b16feb4d61cd6a5526a66962bf3ae83/containers/grizzly2-http/src/main/java/org/glassfish/jersey/grizzly2/httpserver/GrizzlyHttpContainer.java#L256-L281,
479    // https://github.com/eclipse-ee4j/jersey/blob/b7fbb3e75b16feb4d61cd6a5526a66962bf3ae83/containers/jdk-http/src/main/java/org/glassfish/jersey/jdkhttp/JdkHttpHandlerContainer.java#L301-L311,
480    // https://github.com/eclipse-ee4j/jersey/blob/b7fbb3e75b16feb4d61cd6a5526a66962bf3ae83/containers/jetty-http/src/main/java/org/glassfish/jersey/jetty/JettyHttpContainer.java#L332-L357,
481    // https://github.com/eclipse-ee4j/jersey/blob/eafb9bdcb82dfa3fd76dd957d307b99d4a22c87f/containers/jersey-servlet-core/src/main/java/org/glassfish/jersey/servlet/internal/ResponseWriter.java#L212-L238,
482    // and
483    // https://github.com/eclipse-ee4j/jersey/blob/eafb9bdcb82dfa3fd76dd957d307b99d4a22c87f/containers/simple-http/src/main/java/org/glassfish/jersey/simple/SimpleContainer.java#L219-L232.
484    //
485    // All these Jersey-supplied containers rethrow the supplied
486    // Throwable. But the Netty one does not:
487    // https://github.com/eclipse-ee4j/jersey/blob/eafb9bdcb82dfa3fd76dd957d307b99d4a22c87f/containers/netty-http/src/main/java/org/glassfish/jersey/netty/httpserver/NettyResponseWriter.java#L177-L181.
488    //
489    // See also https://github.com/jersey/jersey/pull/3791, where it
490    // is implied that the fact that Jersey's own Netty integration
491    // does NOT rethrow it is a bug.  There is some ambiguity about
492    // whether the failure(Throwable) method (this method) is supposed
493    // to fully handle the supplied Throwable or if it is supposed to
494    // propagate it outwards.  The ambiguity arises because you could
495    // see that this very method is the place where an unhandled
496    // Throwable is propagated--after all, we're in a
497    // ContainerResponseWriter implementation--or you could see that
498    // whatever is housing this class is the container, so the
499    // supplied Throwable should be wrapped and rethrown.  JAX-RS
500    // section 3.3.4 is not very helpful here.  We follow the other
501    // Jersey-supplied containers here.
502    if (throwable instanceof RuntimeException) {
503      throw (RuntimeException)throwable;
504    } else if (throwable instanceof Exception) {
505      throw new ContainerException(throwable.getMessage(), throwable);
506    } else if (throwable instanceof Error) {
507      throw (Error)throwable;
508    } else {
509      throw new InternalError(throwable.getMessage(), throwable);
510    }
511  }

539  @Override
540  public final boolean suspend(final long timeout, final TimeUnit timeUnit, final TimeoutHandler timeoutHandler) {
541    assert !this.inEventLoop();
542
543    // Lifted from Jersey's supplied Netty integration, with repairs.
544    final boolean returnValue;
545    if (this.suspendTimeoutHandler != null) {
546      returnValue = false;
547    } else {
548      this.suspendTimeoutHandler = () -> {
549        timeoutHandler.onTimeout(this);
550        // TODO: not sure about this
551        this.suspendTimeoutHandler = null;
552      };
553      if (timeout > 0) {
554        this.suspendTimeoutFuture =
555          this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
556      }
557      returnValue = true;
558    }
559    return returnValue;
560  }
561
562  @Override
563  public final void setSuspendTimeout(final long timeout, final TimeUnit timeUnit) {
564    assert !this.inEventLoop();
565
566    // Lifted from Jersey's supplied Netty integration, with repairs.
567    if (this.suspendTimeoutHandler == null) {
568      throw new IllegalStateException();
569    }
570    if (this.suspendTimeoutFuture != null) {
571      this.suspendTimeoutFuture.cancel(true);
572    }
573    if (timeout > 0) {
574      this.suspendTimeoutFuture =
575        this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
576    }
577  }

740  /**
741   * A utility function that copies entries from a source {@link Map}
742   * by passing each entry to the supplied {@link BiConsumer},
743   * transforming the keys beforehand using the supplied {@link
744   * UnaryOperator} and that is intended in this framework to be used
745   * to copy HTTP or HTTP/2 headers to and from the proper places.
746   *
747   * @param headersSource the source of the headers to copy; may be
748   * {@code null} in which case no action will be taken
749   *
750   * @param keyTransformer a {@link UnaryOperator} that transforms a
751   * header name; if {@code null} then the return value of {@link
752   * UnaryOperator#identity()} will be used instead
753   *
754   * @param headersTarget where the headers will be copied to; may be
755   * {@code null} in which case no action will be taken
756   *
757   * @see #writeAndFlushStatusAndHeaders(ContainerResponse, long,
758   * ChannelPromise)
759   */
760  public static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
761                                       UnaryOperator<String> keyTransformer,
762                                       final BiConsumer<? super String, ? super List<String>> headersTarget) {
763    if (headersTarget != null && headersSource != null && !headersSource.isEmpty()) {
764      final Collection<? extends Entry<? extends String, ? extends List<String>>> entrySet = headersSource.entrySet();
765      if (entrySet != null && !entrySet.isEmpty()) {
766        if (keyTransformer == null) {
767          keyTransformer = UnaryOperator.identity();
768        }
769        for (final Entry<? extends String, ? extends List<String>> entry : entrySet) {
770          if (entry != null) {
771            headersTarget.accept(keyTransformer.apply(entry.getKey()), entry.getValue());
772          }
773        }
774      }
775    }
776  }
