142  @Override
143  public final OutputStream writeResponseStatusAndHeaders(final long contentLength, final ContainerResponse containerResponse) throws ContainerException {
144    Objects.requireNonNull(containerResponse);
145    assert !this.inEventLoop();
146
147    this.writeAndFlushStatusAndHeaders(containerResponse, contentLength);
148    
149    final OutputStream returnValue;
150    if (this.needsOutputStream(contentLength)) {
151      assert contentLength != 0L;
152      
153      // We've determined that there is a payload/entity.  Our
154      // ultimate responsibility is to return an OutputStream the
155      // end user's Jersey resource class will write to.
156      //
157      // Allocate a ByteBuf suitable for doing IO.  This could be
158      // heap-based or native.  We don't care; we trust Netty.
159      final ByteBuf byteBuf;
160      if (contentLength > 0L && contentLength <= Integer.MAX_VALUE) {
161        // Positive content length.
162        byteBuf = this.channelHandlerContext.alloc().ioBuffer((int)contentLength);
163      } else {
164        // Negative content length or ridiculously huge content
165        // length so ignore it for capacity purposes.
166        byteBuf = this.channelHandlerContext.alloc().ioBuffer();
167      }
168      assert byteBuf != null;
169      
170      // We store a reference to it ONLY so that we can release it
171      // at #commit() or #failure(Throwable) time.
172      this.byteBuf = byteBuf;
173      
174      // A ChunkedInput despite its name has nothing to do with
175      // chunked Transfer-Encoding.  It is usable anywhere you like.
176      // So here, because writes are happening on an OutputStream in
177      // Jersey-land on a separate thread, and are then being run on
178      // the event loop, unless we were to block the event loop and
179      // wait for the OutputStream to close, we wouldn't know when
180      // to do the write.  So we use ChunkedInput even in cases
181      // where the Content-Length is set to a positive integer.
182      final ChunkedInput<?> chunkedInput = this.createChunkedInput(this.channelHandlerContext.executor(), byteBuf, contentLength);
183      if (chunkedInput == null) {
184        this.byteBuf = null;
185        byteBuf.release();
186        throw new IllegalStateException("createChunkedInput() == null");
187      }
188      this.chunkedInput = chunkedInput;
189      
190      // Enqueue a task that will query the ChunkedInput for
191      // its chunks via its readChunk() method on the event loop.
192      channelHandlerContext.write(this.chunkedInput);
193      
194      // Then return an OutputStream implementation that writes to
195      // the very same ByteBuf, ensuring that writes take place on
196      // the event loop.  The net effect is that as this stream
197      // writes to the ByteBuf, the ChunkedWriteHandler consuming
198      // the ChunkedInput by way of its readChunk() method, also on
199      // the event loop, will stream the results as they are made
200      // available.
201      //
202      // TODO: replace the null third parameter with a
203      // GenericFutureListener that will Do The Right Thing™ with
204      // any exceptions.  **Remember that this listener will be
205      // invoked on the event loop.**
206      returnValue =
207        new EventLoopPinnedByteBufOutputStream(this.channelHandlerContext.executor(),
208                                               byteBuf,
209                                               null);
210    } else {
211      returnValue = null;
212    }
213    return returnValue;
214  }

216  /**
217   * Invoked on a non-event-loop thread by Jersey when, as far as
218   * Jersey is concerned, all processing has completed successfully.
219   *
220   * <p>This implementation ensures that {@link ChunkedInput#close()}
221   * is called on the Netty event loop on the {@link ChunkedInput}
222   * returned by the {@link #createChunkedInput(EventExecutor,
223   * ByteBuf, long)} method, and that a {@linkplain
224   * #writeLastContentMessage() final content message is written
225   * immediately afterwards}, and that the {@link
226   * ChannelHandlerContext} is {@linkplain
227   * ChannelHandlerContext#flush() flushed}.  All of these invocations
228   * occur on the event loop.</p>
229   *
230   * @see ChunkedInput#close()
231   *
232   * @see #createChunkedInput(EventExecutor, ByteBuf, long)
233   *
234   * @see #writeLastContentMessage()
235   *
236   * @see ChannelHandlerContext#write(Object)
237   *
238   * @see ChannelHandlerContext#flush()
239   *
240   * @see ContainerResponseWriter#commit()
241   */
242  @Override
243  public final void commit() {
244    assert !this.inEventLoop();
245    final ChunkedInput<?> chunkedInput = this.chunkedInput;
246    this.chunkedInput = null;
247    final ReferenceCounted byteBuf = this.byteBuf;
248    this.byteBuf = null;
249    if (chunkedInput != null) {
250      this.channelHandlerContext.executor().submit((Callable<Void>)() -> {
251          assert inEventLoop();
252          assert chunkedInput != null;
253          assert byteBuf != null;
254          assert byteBuf.refCnt() == 1;
255            
256          // Mark our ChunkedInput as *closed to new input*.  It may
257          // still contain contents that need to be read.  (This
258          // effectively flips an internal switch in the ChunkedInput
259          // that allows for isEndOfInput() to one day return true.)
260          chunkedInput.close();
261          
262          // This will tell the ChunkedWriteHandler outbound from us
263          // to "drain" our ChunkedInput via successive calls
264          // to #readChunk(ByteBufAllocator).  The isEndOfInput()
265          // method will be called as part of this process and will
266          // eventually return true.
267          channelHandlerContext.flush();
268          
269          // Assert that the ChunkedInput is drained and release its
270          // wrapped ByteBuf.
271          assert chunkedInput.isEndOfInput();
272          final boolean released = byteBuf.release();
273          assert released;
274          
275          // Send whatever magic message it is that tells the HTTP
276          // machinery to finish up.
277          this.writeLastContentMessage();        
278          return null;
279        });
280    }
281  }

283  /**
284   * Invoked on a non-event-loop thread by Jersey when, as far as
285   * Jersey is concerned, all processing has completed unsuccessfully.
286   *
287   * @param throwable the {@link Throwable} that caused failure;
288   * strictly speaking may be {@code null}
289   *
290   * @exception RuntimeException if an error occurs while processing
291   * the supplied {@link Throwable}; the supplied {@link Throwable}
292   * will be {@linkplain Throwable#addSuppressed(Throwable) added as a
293   * suppressed <code>Throwable</code>}
294   *
295   * @exception Error if an error occurs while processing the supplied
296   * {@link Throwable}; the supplied {@link Throwable} will be
297   * {@linkplain Throwable#addSuppressed(Throwable) added as a
298   * suppressed <code>Throwable</code>}
299   *
300   * @see ContainerResponseWriter#failure(Throwable)
301   */
302  @Override
303  public final void failure(final Throwable throwable) {
304    assert !this.inEventLoop();
305
306    try {
307      final ChunkedInput<?> chunkedInput = this.chunkedInput;
308      this.chunkedInput = null;
309      final ReferenceCounted byteBuf = this.byteBuf;
310      this.byteBuf = null;
311      if (chunkedInput != null) {
312        this.channelHandlerContext.executor().submit(() -> {
313            try {
314              assert inEventLoop();
315              this.writeFailureMessage();
316              assert byteBuf != null;
317              assert byteBuf.refCnt() == 1;
318              final boolean released = byteBuf.release();
319              assert released;
320            } catch (final RuntimeException | Error throwMe) {
321              if (throwable != null) {
322                throwMe.addSuppressed(throwable);
323              }
324              throw throwMe;
325            } finally {
326              channelHandlerContext.flush();
327              channelHandlerContext.close();
328            }
329            return null;
330          });
331      }
332    } catch (final RuntimeException | Error throwMe) {
333      if (throwable != null) {
334        throwMe.addSuppressed(throwable);
335      }
336      throw throwMe;
337    }
338  }

366  @Override
367  public final boolean suspend(final long timeout, final TimeUnit timeUnit, final TimeoutHandler timeoutHandler) {
368    assert !this.inEventLoop();
369
370    // Lifted from Jersey's supplied Netty integration, with repairs.
371    final boolean returnValue;
372    if (this.suspendTimeoutHandler != null) {
373      returnValue = false;
374    } else {
375      this.suspendTimeoutHandler = () -> {
376        timeoutHandler.onTimeout(this);
377        // TODO: not sure about this
378        this.suspendTimeoutHandler = null;
379      };
380      if (timeout > 0) {
381        this.suspendTimeoutFuture =
382          this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
383      }
384      returnValue = true;
385    }
386    return returnValue;
387  }
388
389  @Override
390  public final void setSuspendTimeout(final long timeout, final TimeUnit timeUnit) {
391    assert !this.inEventLoop();
392
393    // Lifted from Jersey's supplied Netty integration, with repairs.
394    if (this.suspendTimeoutHandler == null) {
395      throw new IllegalStateException();
396    }
397    if (this.suspendTimeoutFuture != null) {
398      this.suspendTimeoutFuture.cancel(true);
399    }
400    if (timeout > 0) {
401      this.suspendTimeoutFuture =
402        this.scheduledExecutorServiceSupplier.get().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
403    }
404  }

453  /**

456   * ContainerResponse} being processed.
457   *
458   * <p>Implementations of this method must not return {@code
459   * null}.</p>
460   *
461   * <p>Implementations of this method will be called only after
462   * {@link #needsOutputStream(long)} has returned {@code true}.</p>

469   * supplied {@link ByteBuf}.  Undefined behavior will result if this
470   * is not the case.</p>

473   * convenience in case any operations need to be performed on the

477   * @param source the {@link ByteBuf} serving as the source of data;
478   * must not be {@code null}; must only be read from, not written to

480   * @param contentLength the overall length of the content, in bytes,

520  public static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
521                                       final BiConsumer<? super String, ? super List<String>> headersTarget) {
522    copyHeaders(headersSource, UnaryOperator.identity(), headersTarget);
523  }
524  
525  public static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
526                                       UnaryOperator<String> keyTransformer,
527                                       final BiConsumer<? super String, ? super List<String>> headersTarget) {
528    if (headersTarget != null && headersSource != null && !headersSource.isEmpty()) {
529      final Collection<? extends Entry<? extends String, ? extends List<String>>> entrySet = headersSource.entrySet();
530      if (entrySet != null && !entrySet.isEmpty()) {
531        if (keyTransformer == null) {
532          keyTransformer = UnaryOperator.identity();
533        }
534        for (final Entry<? extends String, ? extends List<String>> entry : entrySet) {
535          if (entry != null) {
536            headersTarget.accept(keyTransformer.apply(entry.getKey()), entry.getValue());
537          }
538        }
539      }
540    }
