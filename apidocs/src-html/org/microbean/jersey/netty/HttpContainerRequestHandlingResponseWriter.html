057/**
058 * An {@link AbstractContainerRequestHandlingResponseWriter}
059 * implemented in terms of {@link HttpRequest}s, {@link HttpResponse}s
060 * and {@link
061 * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}s.
062 *
063 * @author <a href="https://about.me/lairdnelson" target="_parent">Laird Nelson</a>
064 *
065 * @see AbstractContainerRequestHandlingResponseWriter
066 *
067 * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
068 *
069 * @see #writeStatusAndHeaders(long, ContainerResponse)
070 *
071 * @see #createOutputStream(long, ContainerResponse)
072 */
073public final class HttpContainerRequestHandlingResponseWriter extends AbstractContainerRequestHandlingResponseWriter<HttpContent> {
074

085  private static final GenericFutureListener<? extends Future<? super Void>> listener = f -> {
086    final Throwable cause = f.cause();
087    if (cause != null && logger.isLoggable(Level.SEVERE)) {
088      logger.log(Level.SEVERE, cause.getMessage(), cause);
089    }    
090  };

141   * whose {@link ApplicationHandler#handle(ContainerRequest)} method
142   * will serve as the bridge between Netty and Jersey; may be {@code
143   * null} somewhat pathologically but normally is not
144   *
145   * @param flushThreshold the minimum number of bytes that an {@link
146   * OutputStream} returned by the {@link #createOutputStream(long,
147   * ContainerResponse)} method must write before an automatic
148   * {@linkplain OutputStream#flush() flush} may take place; if less

151   * flushing will occur

154   * by the {@link #createOutputStream(long, ContainerResponse)}

157   * @see ApplicationHandler
158   *
159   * @see ApplicationHandler#handle(ContainerRequest)

162                                                    final int flushThreshold,
163                                                    final ByteBufCreator byteBufCreator) {
164    super(applicationHandler, flushThreshold, byteBufCreator);
165  }

174  @Override
175  protected final boolean writeStatusAndHeaders(final long contentLength,
176                                                final ContainerResponse containerResponse) {
177    Objects.requireNonNull(containerResponse);
178    
179    final ContainerRequest containerRequest = containerResponse.getRequestContext();
180    if (containerRequest == null) {
181      throw new IllegalArgumentException("containerResponse.getRequestContext() == null");
182    }
183    
184    final Object httpRequestValue = containerRequest.getProperty(HttpRequest.class.getName());
185    if (!(httpRequestValue instanceof HttpRequest)) {
186      throw new IllegalArgumentException("containerResponse; !(containerResponse.getRequestContext().getProperty(\"" +
187                                         HttpRequest.class.getName() +
188                                         "\") instanceof HttpRequest): " + httpRequestValue);
189    }
190    final HttpRequest httpRequest = (HttpRequest)httpRequestValue;
191    
192    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
193
194    final HttpVersion httpVersion = httpRequest.protocolVersion();
195    assert httpVersion != null;
196    this.httpVersion = httpVersion;
197
198    final HttpResponseStatus status;
199    final String reasonPhrase = containerResponse.getStatusInfo().getReasonPhrase();
200    if (reasonPhrase == null) {
201      status = HttpResponseStatus.valueOf(containerResponse.getStatus());
202    } else {
203      status = HttpResponseStatus.valueOf(containerResponse.getStatus(), reasonPhrase);
204    }
205
206    final HttpMessage httpResponse;
207    final boolean needsOutputStream;
208    if (contentLength < 0L) {
209      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerRequest.getMethod());
210      httpResponse = new DefaultHttpResponse(httpVersion, status);
211      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
212      HttpUtil.setTransferEncodingChunked(httpResponse, true);
213    } else if (contentLength == 0L) {
214      needsOutputStream = false;
215      httpResponse = new DefaultFullHttpResponse(httpVersion, status);
216      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
217      HttpUtil.setContentLength(httpResponse, 0L);
218    } else {
219      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerRequest.getMethod());
220      httpResponse = new DefaultHttpResponse(httpVersion, status);
221      copyHeaders(containerResponse.getStringHeaders(), httpResponse.headers());
222      HttpUtil.setContentLength(httpResponse, contentLength);
223    }
224    if (HttpUtil.isKeepAlive(httpRequest)) {
225      HttpUtil.setKeepAlive(httpResponse, true);
226    }
227    
228    final ChannelPromise channelPromise = channelHandlerContext.newPromise();
229    assert channelPromise != null;
230    channelPromise.addListener(listener);
231    
232    if (needsOutputStream) {
233      channelHandlerContext.write(httpResponse, channelPromise);
234    } else {
235      channelHandlerContext.writeAndFlush(httpResponse, channelPromise);
236    }
237
238    return needsOutputStream;
239  }

241  /**
242   * Creates and returns a new {@link
243   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}.
244   *
245   * @param contentLength the content length as determined by the
246   * logic encapsulated by the {@link
247   * ApplicationHandler#handle(ContainerRequest)} method; must not be
248   * {@code 0L}
249   *
250   * @param containerResponse a {@link ContainerResponse} for which an
251   * {@link OutputStream} is being created and returned; must not be
252   * {@code null}; ignored by this implementation
253   *
254   * @return a new {@link
255   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}
256   *
257   * @exception NullPointerException if {@code containerResponse} is
258   * {@code null} or if {@link #getChannelHandlerContext()} returns
259   * {@code null}
260   *
261   * @exception IllegalArgumentException if {@code contentLength} is
262   * {@code 0L}
263   *
264   * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
265   */
266  @Override
267  protected AbstractChannelOutboundInvokingOutputStream<? extends HttpContent> createOutputStream(final long contentLength,
268                                                                                                  final ContainerResponse containerResponse) {
269    Objects.requireNonNull(containerResponse);
270    if (contentLength == 0L) {
271      throw new IllegalArgumentException("contentLength == 0L");
272    }
273    final AbstractChannelOutboundInvokingOutputStream<? extends HttpContent> returnValue =
274      new ByteBufBackedChannelOutboundInvokingHttpContentOutputStream(this.getChannelHandlerContext(),
275                                                                      this.getFlushThreshold(),
276                                                                      false,
277                                                                      this.getByteBufCreator());
278    return returnValue;
279  }
280
281  /**
282   * Overrides {@link ContainerResponseWriter#commit()} to effectively
283   * do nothing when invoked since the {@link OutputStream} returned
284   * by {@link #createOutputStream(long, ContainerResponse)} handles
285   * committing the response.
286   *
287   * @see #createOutputStream(long, ContainerResponse)
288   */
289  @Override
290  public final void commit() {
291    this.httpVersion = null;
292  }
293
294  /**
295   * Writes an appropriate failure message using the return value of
296   * the {@link #getChannelHandlerContext()} method.
297   *
298   * @param failureCause the {@link Throwable} responsible for this
299   * method's invocation; may be {@code null} in pathological cases;
300   * ignored by this implementation
301   *
302   * @exception NullPointerException if {@link
303   * #getChannelHandlerContext()} returns {@code null}
304   */
305  @Override
306  protected final void writeFailureMessage(final Throwable failureCause) {
307    try {
308      final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
309      final ChannelPromise channelPromise = channelHandlerContext.newPromise();
310      assert channelPromise != null;
311      channelPromise.addListener(listener);
312      final HttpMessage failureMessage = new DefaultFullHttpResponse(this.httpVersion, HttpResponseStatus.INTERNAL_SERVER_ERROR);
313      HttpUtil.setContentLength(failureMessage, 0L);
314      channelHandlerContext.write(failureMessage, channelPromise);
315    } finally {
316      this.httpVersion = null;
317    }
318  }
319
320  private static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
321                                        final HttpHeaders nettyHeaders) {    
322    AbstractContainerRequestHandlingResponseWriter.copyHeaders(headersSource, UnaryOperator.identity(), nettyHeaders::add);
323  }
