033/**
034 * A {@link ChunkedInput} implementation that uses a {@link Function}
035 * to read from a {@link ByteBuf} and return a relevant chunk of its
036 * data.
037 *
038 * @author <a href="https://about.me/lairdnelson"
039 * target="_parent">Laird Nelson</a>
040 *
041 * @see #readChunk(ByteBufAllocator)
042 */
043public class FunctionalByteBufChunkedInput<T> implements BoundedChunkedInput<T> {
044

070   * a {@link ByteBuf} whose {@linkplain ByteBuf#readableBytes()

072   * returned by the {@link #readChunk(ByteBufAllocator)} method.
073   *
074   * <p>This field is never {@code null}.</p>

077   */
078  private final Function<? super ByteBuf, ? extends T> chunkReader;

083   *
084   * <p>This field is never less than {@code -1L}.
085   *
086   * @see #length()
087   */
088  private final long lengthToReport;

090  private volatile boolean noMoreInput;

093   * Indicates that no more input is forthcoming, so assuming other

097   * @see #isEndOfInput()
098   *
099   * @see #close()

109  /**
110   * Creates a new {@link FunctionalByteBufChunkedInput} whose {@link #length()}
111   * method will return {@code -1L}.
112   *
113   * @param byteBuf the {@link ByteBuf} from which chunks will be
114   * {@linkplain #readChunk(ByteBufAllocator) read}; must not be
115   * {@code null}
116   *
117   * @param chunkReader a {@link Function} that will be supplied with
118   * a {@link ByteBuf} whose {@linkplain ByteBuf#readableBytes()
119   * readable bytes} will be used to synthesize a chunk that will be
120   * returned by the {@link #readChunk(ByteBufAllocator)} method; must
121   * not be {@code null}
122   *
123   * @see #FunctionalByteBufChunkedInput(ByteBuf, Function, long)
124   */
125  public FunctionalByteBufChunkedInput(final ByteBuf byteBuf,
126                                       final Function<? super ByteBuf, ? extends T> chunkReader) {
127    this(byteBuf, chunkReader, -1L);
128  }

130  /**
131   * Creates a new {@link FunctionalByteBufChunkedInput}.
132   *
133   * @param byteBuf the {@link ByteBuf} from which chunks will be
134   * {@linkplain #readChunk(ByteBufAllocator) read}; must not be
135   * {@code null}
136   *
137   * @param chunkReader a {@link Function} that will be supplied with
138   * a {@link ByteBuf} whose {@linkplain ByteBuf#readableBytes()
139   * readable bytes} will be used to synthesize a chunk that will be
140   * returned by the {@link #readChunk(ByteBufAllocator)} method; must
141   * not be {@code null}
142   *
143   * @param lengthToReport the length of this {@link ChunkedInput}
144   * implementation in bytes to be reported by the {@link #length()}
145   * method; may be {@code -1L} (and very often is) if a
146   * representation of its length is not available.
147   *
148   * @exception NullPointerException if either {@code byteBuf} or
149   * {@code chunkReader} is {@code null}
150   */
151  public FunctionalByteBufChunkedInput(final ByteBuf byteBuf,
152                                       final Function<? super ByteBuf, ? extends T> chunkReader,
153                                       final long lengthToReport) {
154    super();
155    this.byteBuf = Objects.requireNonNull(byteBuf);
156    this.chunkReader = Objects.requireNonNull(chunkReader);
157    this.lengthToReport = Math.max(-1L, lengthToReport);
158  }

161  /*
162   * Instance methods.
163   */

166  /**
167   * Returns {@code true} if the end of the input has been reached and
168   * subsequent invocations of the {@link
169   * #readChunk(ByteBufAllocator)} method will return {@code
170   * null}.
171   *
172   * @return {@code true} if the end of the input has been reached and
173   * subsequent invocations of the {@link
174   * #readChunk(ByteBufAllocator)} method will return {@code
175   * null}.
176   */
177  @Override
178  public final boolean isEndOfInput() {
179    final String mn = "isEndOfInput";
180    if (logger.isLoggable(Level.FINER)) {
181      logger.entering(cn, mn);
182    }
183
184    // (Don't check this.closed and throw an IllegalStateException
185    // because close() calls this method after setting this.closed to
186    // true.  That ordering is important to prevent
187    // setByteBuf(ByteBuf) from interfering during all this.)
188    
189    final boolean returnValue;
190    if (this.noMoreInput) {
191      ByteBuf byteBuf = this.byteBuf;
192      if (byteBuf == null) {
193        returnValue = true;
194      } else if (byteBuf.refCnt() <= 0) {
195        if (logger.isLoggable(Level.WARNING)) {
196          logger.logp(Level.WARNING, cn, mn, "Unexpected refCnt: {0}", byteBuf.refCnt());
197        }
198        returnValue = true;
199      } else {
200        returnValue = !byteBuf.isReadable();
201        // Ensure that nothing more CAN be written to the byteBuf.
202        byteBuf = byteBuf.asReadOnly();
203        assert byteBuf != null;
204        assert byteBuf.isReadOnly();
205        this.byteBuf = byteBuf;
206      }
207    } else {
208      returnValue = false;
209    }
210
211    if (logger.isLoggable(Level.FINER)) {
212      logger.exiting(cn, mn, Boolean.valueOf(returnValue));
213    }
214    return returnValue;
215  }
216
217  /**
218   * Irrevocably closes this {@link FunctionalByteBufChunkedInput} to
219   * new input.
220   */
221  @Override
222  public final void setEndOfInput() {
223    final String mn = "setEndOfInput";
224    if (logger.isLoggable(Level.FINER)) {
225      logger.entering(cn, mn);
226    }
227    this.noMoreInput = true;
228    if (logger.isLoggable(Level.FINER)) {
229      logger.exiting(cn, mn);
230    }
231  }

233  /**
234   * Calls the {@link #readChunk(ByteBufAllocator)} method and returns
235   * its result.
236   *
237   * <p>This method may and often does return {@code null}.</p>
238   *
239   * @param channelHandlerContext a {@link ChannelHandlerContext}
240   * whose {@link ChannelHandlerContext#alloc()} method is invoked to
241   * acquire a {@link ByteBufAllocator}; may be {@code null}
242   *
243   * @return a chunk of the overall data, or {@code null}
244   *
245   * @see #readChunk(ByteBufAllocator)
246   *
247   * @deprecated Please use the {@link #readChunk(ByteBufAllocator)}
248   * method instead.
249   */
250  @Deprecated
251  @Override
252  public final T readChunk(final ChannelHandlerContext channelHandlerContext) {
253    return this.readChunk(channelHandlerContext == null ? (ByteBufAllocator)null : channelHandlerContext.alloc());
254  }

256  /**
257   * Returns a chunk of this {@link FunctionalByteBufChunkedInput}'s
258   * overall input as represented by the {@link ByteBuf} and using the
259   * {@link Function} supplied to it {@linkplain
260   * #FunctionalByteBufChunkedInput(ByteBuf, Function, long) at
261   * construction time}.
262   *
263   * <p>This method may, and often does, return {@code null}.</p>
264   *
265   * <p>The chunk that is returned by this method will be, if
266   * non-{@code null}, synthesized from a {@linkplain
267   * ByteBuf#readRetainedSlice(int) retained slice} of the {@link
268   * ByteBuf} {@linkplain #FunctionalByteBufChunkedInput(ByteBuf,
269   * Function, long) supplied at construction time} whose size is
270   * given by the return value of the {@link #getChunkSize(ByteBuf)}
271   * method.</p>
272   *
273   * @param ignoredByteBufAllocator a {@link ByteBufAllocator} that an
274   * implementation may use if it wishes but normally should have no
275   * use for; may be {@code null}
276   *
277   * @return a chunk of this {@link ByteBufChunkedInput}'s overall
278   * input, or {@code null}
279   *
280   * @exception IllegalStateException if {@link #close()} has been
281   * called
282   *
283   * @see #getChunkSize(ByteBuf)
284   *
285   * @see #FunctionalByteBufChunkedInput(ByteBuf, Function, long)
286   *
287   * @see #isEndOfInput()
288   *
289   * @see #setEndOfInput()
290   *
291   * @see #close()
292   */
293  @Override
294  public final T readChunk(final ByteBufAllocator ignoredByteBufAllocator) {
295    final String mn = "readChunk";
296    if (logger.isLoggable(Level.FINER)) {
297      logger.entering(cn, mn, ignoredByteBufAllocator);
298    }
299    if (this.closed) {
300      throw new IllegalStateException("closed");
301    }
302
303    final T returnValue;
304    final ByteBuf byteBuf = this.byteBuf;
305    if (byteBuf == null) {
306      returnValue = null;
307    } else if (byteBuf.refCnt() <= 0) {
308      if (logger.isLoggable(Level.WARNING)) {
309        logger.logp(Level.WARNING, cn, mn, "Unexpected refCnt: {0}", byteBuf.refCnt());
310      }
311      returnValue = null;
312    } else if (!byteBuf.isReadable()) {
313      returnValue = null;
314    } else {
315      final int readableBytes = byteBuf.readableBytes();
316      assert readableBytes > 0; // ...because we already checked isReadable()
317      final int numberOfBytesToRead;
318      final int chunkSize = this.getChunkSize(byteBuf);
319      if (readableBytes < chunkSize) {
320        numberOfBytesToRead = readableBytes;
321      } else {
322        numberOfBytesToRead = chunkSize;
323      }
324      // TODO: look REALLY hard at whether this slice should be retained or not.
325      final ByteBuf retainedSlice = byteBuf.readRetainedSlice(numberOfBytesToRead);
326      assert retainedSlice != null;
327      returnValue = this.chunkReader.apply(retainedSlice.asReadOnly());
328    }
329
330    if (logger.isLoggable(Level.FINER)) {
331      logger.exiting(cn, mn, returnValue);
332    }
333    return returnValue;
334  }

336  /**
337   * Returns the size of the chunk, in bytes, that will be read and
338   * synthesized into the return value the {@link
339   * #readChunk(ByteBufAllocator)} method.
340   *
341   * <p>This implementation returns the result of invoking {@code
342   * source.}{@link ByteBuf#readableBytes() readableBytes()} on the
343   * supplied {@code source}.</p>
344   *
345   * @param source the {@link ByteBuf} that was {@linkplain
346   * #FunctionalByteBufChunkedInput(ByteBuf, Function, long) supplied
347   * at construction time}; may be {@code null} in which case {@code
348   * 0} will be returned
349   *
350   * @return the size of the chunk, in bytes, that will be returned by
351   * the {@link #readChunk(ByteBufAllocator)} method; <strong>behavior
352   * is undefined if this value is less than or equal to {@code
353   * 0}</strong>
354   *
355   * @exception NullPointerException if for any reason {@code source}
356   * is {@code null}
357   *
358   * @see #readChunk(ByteBufAllocator)
359   */
360  protected int getChunkSize(final ByteBuf source) {
361    return source == null ? 0 : source.readableBytes();
362  }

364  /**
365   * Returns the length of this {@link ChunkedInput} implementation in bytes,
366   * or {@code -1L} if a representation of its length is not
367   * available.
368   *
369   * @return the length of this {@link ChunkedInput} implementation in
370   * bytes, or {@code -1L}
371   *
372   * @exception IllegalStateException if {@link #close()} has been
373   * called
374   */
375  @Override
376  public final long length() {
377    if (this.closed) {
378      throw new IllegalStateException("closed");
379    }
380    return this.lengthToReport;
381  }
382
383  /**
384   * Returns the number of bytes read from this input.
385   *
386   * <p>If the return value of an invocation of the {@link #length()}
387   * method returns zero or a positive integer, then the value
388   * returned by an invocation of this method will be less than or
389   * equal to that value.</p>
390   *
391   * @return the number of bytes read from this input
392   *
393   * @exception IllegalStateException if {@link #close()} has been
394   * called
395   *
396   * @see #length()
397   */
398  @Override
399  public final long progress() {
400    if (this.closed) {
401      throw new IllegalStateException("closed");
402    }
403    // e.g. we've read <progress> of <length> bytes.  Other
404    // ChunkedInput implementations return a valid number here even
405    // when length() returns -1L, so we do too.
406    final long returnValue;
407    final ByteBuf byteBuf = this.byteBuf;
408    if (byteBuf == null) {
409      returnValue = 0L;
410    } else if (byteBuf.refCnt() <= 0) {
411      if (logger.isLoggable(Level.WARNING)) {
412        logger.logp(Level.WARNING, cn, "progress", "Unexpected refCnt: {0}", byteBuf.refCnt());
413      }
414      returnValue = 0L;
415    } else {
416      returnValue = byteBuf.readerIndex();
417    }
418    return returnValue;
419  }
420
421  /**
422   * Marks this {@link FunctionalByteBufChunkedInput} as being closed.
423   *
424   * <p>As a result the {@link #readChunk(ByteBufAllocator)} method
425   * may return {@code null} in the future.</p>
426   *
427   * @see #readChunk(ByteBufAllocator)
428   */
429  @Override
430  public final void close() {
431    final String mn = "close";
432    if (logger.isLoggable(Level.FINER)) {
433      logger.entering(cn, mn);
434    }
435    if (this.closed) {
436      throw new IllegalStateException("closed");
437    }
438    
439    this.closed = true;
440    this.setEndOfInput();
441    final ByteBuf byteBuf = this.byteBuf;
442    if (byteBuf != null) {
443      if (byteBuf.refCnt() <= 0) {
444        if (logger.isLoggable(Level.WARNING)) {
445          logger.logp(Level.WARNING, cn, mn, "Unexpected refCnt: {0}", byteBuf.refCnt());
446        }
447      } else {
448        assert byteBuf.isReadOnly();
449        byteBuf.release();
450      }
451    }
452    this.byteBuf = null;
453
454    if (logger.isLoggable(Level.FINER)) {
455      logger.exiting(cn, mn);
456    }
457  }
