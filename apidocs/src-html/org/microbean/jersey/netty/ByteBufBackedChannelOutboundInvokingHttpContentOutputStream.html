017package org.microbean.jersey.netty;
018
019import io.netty.buffer.ByteBuf;
020
021import io.netty.channel.ChannelOutboundInvoker;
022
023import io.netty.handler.codec.http.DefaultLastHttpContent;
024import io.netty.handler.codec.http.DefaultHttpContent;
025import io.netty.handler.codec.http.HttpContent;
026
027/**
028 * An {@link AbstractByteBufBackedChannelOutboundInvokingOutputStream}
029 * that writes {@link HttpContent} messages.
030 *
031 * @author <a href="https://about.me/lairdnelson"
032 * target="_parent">Laird Nelson</a>
033 *
034 * @see #createMessage(ByteBuf)
035 *
036 * @see #createLastMessage()
037 */
038public final class ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
039  extends AbstractByteBufBackedChannelOutboundInvokingOutputStream<HttpContent> {
040
041
042  /*
043   * Constructors.
044   */
045
046
047  /**
048   * Creates a new {@link
049   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}.
050   *
051   * @param channelOutboundInvoker the {@link ChannelOutboundInvoker}
052   * to which operations are adapted; must not be {@code null}
053   *
054   * @param closeChannelOutboundInvoker whether {@link
055   * ChannelOutboundInvoker#close(ChannelPromise)} will be called on
056   * the supplied {@link ChannelOutboundInvoker} when {@link #close()
057   * close()} is called
058   *
059   * @see
060   * #ByteBufBackedChannelOutboundInvokingHttpContentOutputStream(ChannelOutboundInvoker,
061   * int, boolean,
062   * AbstractByteBufBackedChannelOutboundInvokingOutputStream.ByteBufCreator)
063   */
064  public ByteBufBackedChannelOutboundInvokingHttpContentOutputStream(final ChannelOutboundInvoker channelOutboundInvoker,
065                                                                     final boolean closeChannelOutboundInvoker) {
066    this(channelOutboundInvoker, Integer.MAX_VALUE, closeChannelOutboundInvoker, null);
067  }
068
069  /**
070   * Creates a new {@link
071   * ByteBufBackedChannelOutboundInvokingHttpContentOutputStream}.
072   *
073   * @param channelOutboundInvoker the {@link ChannelOutboundInvoker}
074   * to which operations are adapted; must not be {@code null}
075   *
076   * @param flushThreshold the minimum number of bytes that this
077   * instance has to {@linkplain #write(byte[], int, int) write}
078   * before an automatic {@linkplain #flush() flush} will take place;
079   * if less than {@code 0} {@code 0} will be used instead; if {@code
080   * Integer#MAX_VALUE} then no automatic flushing will occur
081   *
082   * @param closeChannelOutboundInvoker whether {@link
083   * ChannelOutboundInvoker#close(ChannelPromise)} will be called on
084   * the supplied {@link ChannelOutboundInvoker} when {@link #close()
085   * close()} is called
086   *
087   * @param byteBufCreator a {@link ByteBufCreator} that will be used
088   * to {@linkplain ByteBufCreator#toByteBuf(byte[], int, int) create
089   * <code>ByteBuf</code> instances}; may be {@code null} in which
090   * case a default {@link ByteBufCreator} adapting {@link
091   * io.netty.buffer.Unpooled#wrappedBuffer(byte[], int, int)} will be
092   * used instead
093   */
094  public ByteBufBackedChannelOutboundInvokingHttpContentOutputStream(final ChannelOutboundInvoker channelOutboundInvoker,
095                                                                     final int flushThreshold,
096                                                                     final boolean closeChannelOutboundInvoker,
097                                                                     final ByteBufCreator byteBufCreator) {
098    super(channelOutboundInvoker, flushThreshold, closeChannelOutboundInvoker, byteBufCreator);
099  }
100
101
102  /*
103   * Instance methods.
104   */
105
106  
107  /**
108   * Returns a new {@link DefaultHttpContent} whose {@link
109   * DefaultHttpContent#content() content()} method returns the
110   * supplied {@link ByteBuf}.
111   *
112   * <p>This method never returns {@code null}.</p>
113   *
114   * @param content a {@link ByteBuf}; must not be {@code null}
115   *
116   * @return a new {@link DefaultHttpContent} whose {@link
117   * DefaultHttpContent#content() content()} method returns the
118   * supplied {@link ByteBuf}; never {@code null}
119   */
120  @Override
121  protected final HttpContent createMessage(final ByteBuf content) {
122    return new DefaultHttpContent(content);
123  }
124
125
126  /**
127   * Returns a new {@link DefaultLastHttpContent} when invoked.
128   *
129   * <p>This method never returns {@code null}.</p>
130   *
131   * @return a new {@link DefaultLastHttpContent}
132   *
133   * @see #close()
134   */
135  @Override
136  protected final HttpContent createLastMessage() {
137    return new DefaultLastHttpContent();
138  }
139
140}
