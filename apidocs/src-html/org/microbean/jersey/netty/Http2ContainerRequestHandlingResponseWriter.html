050/**
051 * An {@link AbstractContainerRequestHandlingResponseWriter}
052 * implemented in terms of {@link Http2Headers}, {@link
053 * Http2HeadersFrame}s, {@link Http2DataFrame}s and {@link
054 * ByteBufBackedChannelOutboundInvokingHttp2DataFrameOutputStream}s.
055 *
056 * @author <a href="https://about.me/lairdnelson" target="_parent">Laird Nelson</a>
057 *
058 * @see AbstractContainerRequestHandlingResponseWriter
059 *
060 * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
061 *
062 * @see #writeStatusAndHeaders(long, ContainerResponse)
063 *
064 * @see #createOutputStream(long, ContainerResponse)
065 */
066public final class Http2ContainerRequestHandlingResponseWriter extends AbstractContainerRequestHandlingResponseWriter<Http2DataFrame> {

113   * will serve as the bridge between Netty and Jersey; may be {@code
114   * null} somewhat pathologically but normally is not
115   *
116   * @param flushThreshold the minimum number of bytes that an {@link
117   * AbstractChannelOutboundInvokingOutputStream} returned by the
118   * {@link #createOutputStream(long, ContainerResponse)} method must

136    super(applicationHandler, flushThreshold, byteBufCreator);
137  }

140  /*
141   * Instance methods.
142   */

145  /**
146   * Writes the status and headers portion of the response present in
147   * the supplied {@link ContainerResponse} and returns {@code true}
148   * if further output is forthcoming.
149   *
150   * <p>This implementation writes an instance of {@link
151   * DefaultHttp2HeadersFrame}.</p>
152   *
153   * @param contentLength the content length as determined by the
154   * logic encapsulated by the {@link
155   * ApplicationHandler#handle(ContainerRequest)} method; a value less
156   * than zero indicates an unknown content length
157   *
158   * @param containerResponse the {@link ContainerResponse} containing
159   * status and headers information; must not be {@code null}
160   *
161   * @return {@code true} if the {@link #createOutputStream(long,
162   * ContainerResponse)} method should be invoked, <em>i.e.</em> if
163   * further output is forthcoming
164   *
165   * @exception NullPointerException if {@code containerResponse} is
166   * {@code null}
167   *
168   * @see ApplicationHandler#handle(ContainerRequest)
169   *
170   * @see #createOutputStream(long, ContainerResponse)
171   */
172  @Override
173  protected final boolean writeStatusAndHeaders(final long contentLength,
174                                                final ContainerResponse containerResponse) {
175    Objects.requireNonNull(containerResponse);
176
177    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
178
179    final Http2Headers nettyHeaders = new DefaultHttp2Headers();
180    copyHeaders(containerResponse.getStringHeaders(), String::toLowerCase, nettyHeaders::add);
181    // See https://tools.ietf.org/html/rfc7540#section-8.1.2.4
182    nettyHeaders.status(Integer.toString(containerResponse.getStatus()));
183
184    final boolean needsOutputStream;
185    if (contentLength < 0L) {
186      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerResponse.getRequestContext().getMethod());
187      // Don't set the Content-Length: header because the content
188      // length is unknown.
189    } else if (contentLength == 0L) {
190      needsOutputStream = false;
191      nettyHeaders.set(HttpHeaders.CONTENT_LENGTH.toLowerCase(), "0");
192    } else {
193      needsOutputStream = !HttpMethod.HEAD.equalsIgnoreCase(containerResponse.getRequestContext().getMethod());
194      nettyHeaders.set(HttpHeaders.CONTENT_LENGTH.toLowerCase(), Long.toString(contentLength));
195    }
196
197    final Object message = new DefaultHttp2HeadersFrame(nettyHeaders, !needsOutputStream /* end of stream? */);
198
199    final ChannelPromise channelPromise = channelHandlerContext.newPromise();
200    assert channelPromise != null;
201    channelPromise.addListener(listener);
202
203    // Remember that
204    // AbstractContainerRequestHandlingResponseWriter#channelReadComplete(ChannelHandlerContext)
205    // will call ChannelHandlerContext#flush() in all cases.
206    channelHandlerContext.write(message, channelPromise);
207
208    return needsOutputStream;
209  }

211  /**
212   * Creates and returns a new {@link
213   * ByteBufBackedChannelOutboundInvokingHttp2DataFrameOutputStream}.
214   *
215   * @param contentLength the content length as determined by the
216   * logic encapsulated by the {@link
217   * ApplicationHandler#handle(ContainerRequest)} method; must not be
218   * {@code 0L}
219   *
220   * @param containerResponse a {@link ContainerResponse} for which an
221   * {@link AbstractChannelOutboundInvokingOutputStream} is being
222   * created and returned; must not be {@code null}; ignored by this
223   * implementation
224   *
225   * @return a new {@link
226   * ByteBufBackedChannelOutboundInvokingHttp2DataFrameOutputStream}
227   *
228   * @exception NullPointerException if {@code containerResponse} is
229   * {@code null} or if {@link #getChannelHandlerContext()} returns
230   * {@code null}
231   *
232   * @exception IllegalArgumentException if {@code contentLength} is
233   * {@code 0L}
234   *
235   * @see ByteBufBackedChannelOutboundInvokingHttpContentOutputStream
236   */
237  @Override
238  protected final AbstractChannelOutboundInvokingOutputStream<? extends Http2DataFrame> createOutputStream(final long contentLength,
239                                                                                                           final ContainerResponse containerResponse) {
240    if (contentLength == 0L) {
241      throw new IllegalArgumentException("contentLength == 0L");
242    }
243    return new ByteBufBackedChannelOutboundInvokingHttp2DataFrameOutputStream(this.getChannelHandlerContext(),
244                                                                              this.getFlushThreshold(),
245                                                                              false,
246                                                                              this.getByteBufCreator()) {
247      @Override
248      protected final ChannelPromise newPromise() {
249        final ChannelPromise returnValue = super.newPromise();
250        if (returnValue != null && !returnValue.isVoid()) {
251          returnValue.addListener(listener);
252        }
253        return returnValue;
254      }
255    };
256  }
257
258  /**
259   * Writes an appropriate failure message using the return value of
260   * the {@link #getChannelHandlerContext()} method.
261   *
262   * @param failureCause the {@link Throwable} responsible for this
263   * method's invocation; may be {@code null} in pathological cases;
264   * ignored by this implementation
265   *
266   * @exception NullPointerException if {@link
267   * #getChannelHandlerContext()} returns {@code null}
268   */
269  @Override
270  protected final void writeFailureMessage(final Throwable failureCause) {
271    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
272    final ChannelPromise channelPromise = channelHandlerContext.newPromise();
273    assert channelPromise != null;
274    channelPromise.addListener(listener);
275    channelHandlerContext.write(new DefaultHttp2Headers().status(String.valueOf(Status.INTERNAL_SERVER_ERROR.getStatusCode())),
276                                channelPromise);
277  }
