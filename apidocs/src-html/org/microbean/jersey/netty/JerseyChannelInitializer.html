017package org.microbean.jersey.netty;

019import java.net.URI;
020
021import java.util.function.BiFunction;
022
023import javax.ws.rs.core.SecurityContext;
024
025import io.netty.buffer.ByteBufAllocator;
026
027import io.netty.channel.Channel;
028import io.netty.channel.ChannelHandler; // for javadoc only
029import io.netty.channel.ChannelHandlerContext;
030import io.netty.channel.ChannelInitializer;
031import io.netty.channel.ChannelPipeline;
032
033import io.netty.handler.codec.http.HttpRequest;
034import io.netty.handler.codec.http.HttpServerCodec;
035import io.netty.handler.codec.http.HttpServerExpectContinueHandler;
036
037import io.netty.handler.logging.LoggingHandler;
038
039import io.netty.handler.ssl.SslContext;
040import io.netty.handler.ssl.SslHandler;
041
042import io.netty.handler.stream.ChunkedWriteHandler;
043
044import org.glassfish.jersey.server.ApplicationHandler;
045
046/**
047 * A {@link ChannelInitializer} that sets up <a
048 * href="https://jersey.github.io/">Jersey</a> integration.
049 *
050 * @author <a href="https://about.me/lairdnelson"
051 * target="_parent">Laird Nelson</a>
052 *
053 * @see ChannelInitializer
054 *
055 * @see #initChannel(Channel)
056 */
057public class JerseyChannelInitializer extends ChannelInitializer<Channel> {
058
059
060  /*
061   * Instance fields.
062   */
063
064  
065  private final URI baseUri;
066
067  private final SslContext sslContext;
068
069  private final ApplicationHandler applicationHandler;
070
071  private final BiFunction<? super ChannelHandlerContext, ? super HttpRequest, ? extends SecurityContext> securityContextBiFunction;
072  
073
074  /*
075   * Constructors.
076   */
077
078
079  /**
080   * Creates a new {@link JerseyChannelInitializer}.
081   *
082   * @param baseUri the base {@link URI} of the Jersey application;
083   * may be {@code null} in which case the return value resulting from
084   * invoking {@link URI#create(String) URI.create("/")} will be used
085   * instead
086   *
087   * @param sslContext an {@link SslContext} that may be used to
088   * {@linkplain #createSslHandler(SslContext, ByteBufAllocator)
089   * create an <code>SslHandler</code>}; may be {@code null}
090   *
091   * @param applicationHandler the {@link ApplicationHandler}
092   * representing Jersey; may be {@code null} in which case a
093   * {@linkplain ApplicationHandler#ApplicationHandler() new
094   * <code>ApplicationHandler</code>} will be used instead
095   *
096   * @param securityContextBiFunction a {@link BiFunction} that
097   * returns a {@link SecurityContext} when supplied with a {@link
098   * ChannelHandlerContext} and an {@link HttpRequest}; may be {@code
099   * null}
100   */
101  public JerseyChannelInitializer(final URI baseUri,
102                                  final SslContext sslContext,
103                                  final ApplicationHandler applicationHandler,
104                                  final BiFunction<? super ChannelHandlerContext, ? super HttpRequest, ? extends SecurityContext> securityContextBiFunction) {
105    super();
106    this.baseUri = baseUri;
107    this.sslContext = sslContext;
108    this.applicationHandler = applicationHandler == null ? new ApplicationHandler() : applicationHandler;
109    this.securityContextBiFunction = securityContextBiFunction;
110  }
111
112
113  /*
114   * Instance methods.
115   */
116  
117
118  /**
119   * Sets up Netty with Jersey application support.
120   *
121   * @param channel the {@link Channel} representing a networking
122   * connection to the outside world; may be {@code null} in which
123   * case no action will be taken
124   *
125   * @see Channel#pipeline()
126   *
127   * @see ChannelPipeline#addLast(String, ChannelHandler)
128   *
129   * @see #preInitChannel(Channel)
130   *
131   * @see #postInitChannel(Channel)
132   */
133  @Override
134  public final void initChannel(final Channel channel) {
135    if (channel != null) {
136      this.preInitChannel(channel);
137      final ChannelPipeline channelPipeline = channel.pipeline();
138      assert channelPipeline != null;
139      if (this.sslContext != null) {
140        final SslHandler sslHandler = createSslHandler(this.sslContext, channel.alloc());
141        if (sslHandler != null) {
142          channelPipeline.addLast(sslHandler.getClass().getSimpleName(), sslHandler);
143        }
144      }
145      channelPipeline.addLast(HttpServerCodec.class.getSimpleName(), new HttpServerCodec());
146      channelPipeline.addLast(HttpServerExpectContinueHandler.class.getSimpleName(), new HttpServerExpectContinueHandler());
147      channelPipeline.addLast(ChunkedWriteHandler.class.getSimpleName(), new ChunkedWriteHandler());
148      channelPipeline.addLast(JerseyChannelInboundHandler.class.getSimpleName(), new JerseyChannelInboundHandler(this.baseUri, this.applicationHandler, this.securityContextBiFunction));
149      this.postInitChannel(channel);
150    }
151  }
152
153
154  /**
155   * A hook for performing {@link Channel} initialization before the
156   * Jersey integration is set up.
157   *
158   * <p>This implementation {@linkplain
159   * ChannelPipeline#addLast(String, ChannelHandler) installs} a
160   * {@link LoggingHandler}.</p>
161   *
162   * <p>Overrides must not call {@link #initChannel(Channel)} or an
163   * infinite loop will result.</p>
164   *
165   * @param channel the {@link Channel} being {@linkplain
166   * #initChannel(Channel) initialized}; may be {@code null} in which
167   * case no action will be taken
168   *
169   * @see #initChannel(Channel)
170   *
171   * @see ChannelPipeline#addLast(String, ChannelHandler)
172   */
173  protected void preInitChannel(final Channel channel) {
174    if (channel != null) {
175      final ChannelPipeline channelPipeline = channel.pipeline();
176      assert channelPipeline != null;
177      channelPipeline.addLast("LoggingHandler", new LoggingHandler());
178    }
179  }
180
181  /**
182   * A hook for performing {@link Channel} initialization after the
183   * Jersey integration is set up.
184   *
185   * <p>This implementation does nothing.</p>
186   *
187   * <p>Overrides must not call {@link #initChannel(Channel)} or an
188   * infinite loop will result.</p>
189   *
190   * @param channel the {@link Channel} being {@linkplain
191   * #initChannel(Channel) initialized}; may be {@code null} in which
192   * case no action will be taken
193   *
194   * @see #initChannel(Channel)
195   *
196   * @see ChannelPipeline#addLast(String, ChannelHandler)
197   */
198  protected void postInitChannel(final Channel channel) {
199
200  }
201
202  /**
203   * Creates and returns a new {@link SslHandler} when invoked.
204   *
205   * <p>This method never returns {@code null}.</p>
206   *
207   * <p>Overrides of this method must not return {@code null}.</p>
208   *
209   * @param sslContext the {@link SslContext} that may assist in the
210   * creation; must not be {@code null}
211   *
212   * @param byteBufAllocator a {@link ByteBufAllocator} that may
213   * assist in the creation; must not be {@code null}
214   *
215   * @return a new {@link SslHandler}; never {@code null}
216   *
217   * @see SslContext#newHandler(ByteBufAllocator)
218   */
219  protected SslHandler createSslHandler(final SslContext sslContext, final ByteBufAllocator byteBufAllocator) {
220    return sslContext.newHandler(byteBufAllocator);
221  }
222
223}
