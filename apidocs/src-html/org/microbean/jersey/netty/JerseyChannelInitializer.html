049/**
050 * A {@link ChannelInitializer} that sets up <a
051 * href="https://jersey.github.io/">Jersey</a> integration.
052 *
053 * @author <a href="https://about.me/lairdnelson"
054 * target="_parent">Laird Nelson</a>
055 *
056 * @see ChannelInitializer
057 *
058 * @see #initChannel(Channel)
059 */
060public class JerseyChannelInitializer extends ChannelInitializer<Channel> {
061
062
063  /*
064   * Instance fields.
065   */
066
067  
068  private final URI baseUri;
069
070  private final SslContext sslContext;
071
072  private final ApplicationHandler applicationHandler;
073
074  private final BiFunction<? super ChannelHandlerContext, ? super HttpRequest, ? extends SecurityContext> securityContextBiFunction;
075  
076
077  /*
078   * Constructors.
079   */
080
081
082  /**
083   * Creates a new {@link JerseyChannelInitializer}.
084   *
085   * @param baseUri the base {@link URI} of the Jersey application;
086   * may be {@code null} in which case the return value resulting from
087   * invoking {@link URI#create(String) URI.create("/")} will be used
088   * instead
089   *
090   * @param sslContext an {@link SslContext} that may be used to
091   * {@linkplain #createSslHandler(SslContext, ByteBufAllocator)
092   * create an <code>SslHandler</code>}; may be {@code null}
093   *
094   * @param applicationHandler the {@link ApplicationHandler}
095   * representing Jersey; may be {@code null} in which case a
096   * {@linkplain ApplicationHandler#ApplicationHandler() new
097   * <code>ApplicationHandler</code>} will be used instead
098   *
099   * @param securityContextBiFunction a {@link BiFunction} that
100   * returns a {@link SecurityContext} when supplied with a {@link
101   * ChannelHandlerContext} and an {@link HttpRequest}; may be {@code
102   * null}
103   */
104  public JerseyChannelInitializer(final URI baseUri,
105                                  final SslContext sslContext,
106                                  final ApplicationHandler applicationHandler,
107                                  final BiFunction<? super ChannelHandlerContext, ? super HttpRequest, ? extends SecurityContext> securityContextBiFunction) {
108    super();
109    this.baseUri = baseUri;
110    this.sslContext = sslContext;
111    this.applicationHandler = applicationHandler == null ? new ApplicationHandler() : applicationHandler;
112    this.securityContextBiFunction = securityContextBiFunction;
113  }
114
115
116  /*
117   * Instance methods.
118   */
119  
120
121  /**
122   * Sets up Netty with Jersey application support.
123   *
124   * @param channel the {@link Channel} representing a networking
125   * connection to the outside world; may be {@code null} in which
126   * case no action will be taken
127   *
128   * @see Channel#pipeline()
129   *
130   * @see ChannelPipeline#addLast(String, ChannelHandler)
131   *
132   * @see #preInitChannel(Channel)
133   *
134   * @see #postInitChannel(Channel)
135   */
136  @Override
137  public final void initChannel(final Channel channel) {
138    if (channel != null) {
139      this.preInitChannel(channel);
140      final ChannelPipeline channelPipeline = channel.pipeline();
141      assert channelPipeline != null;
142      if (this.sslContext != null) {
143        final SslHandler sslHandler = createSslHandler(this.sslContext, channel.alloc());
144        if (sslHandler != null) {
145          channelPipeline.addLast(sslHandler.getClass().getSimpleName(), sslHandler);
146        }
147      }
148      channelPipeline.addLast(HttpServerCodec.class.getSimpleName(), new HttpServerCodec());
149      channelPipeline.addLast(HttpServerExpectContinueHandler.class.getSimpleName(), new HttpServerExpectContinueHandler());
150      channelPipeline.addLast(ChunkedWriteHandler.class.getSimpleName(), new ChunkedWriteHandler());
151      channelPipeline.addLast(JerseyChannelInboundHandler.class.getSimpleName(), new JerseyChannelInboundHandler(this.baseUri, this.applicationHandler, this.securityContextBiFunction));
152      this.postInitChannel(channel);
153    }
154  }
155
156
157  /**
158   * A hook for performing {@link Channel} initialization before the
159   * Jersey integration is set up.
160   *
161   * <p>This implementation {@linkplain
162   * ChannelPipeline#addLast(String, ChannelHandler) installs} a
163   * {@link LoggingHandler}.</p>
164   *
165   * <p>Overrides must not call {@link #initChannel(Channel)} or an
166   * infinite loop will result.</p>
167   *
168   * @param channel the {@link Channel} being {@linkplain
169   * #initChannel(Channel) initialized}; may be {@code null} in which
170   * case no action will be taken
171   *
172   * @see #initChannel(Channel)
173   *
174   * @see ChannelPipeline#addLast(String, ChannelHandler)
175   */
176  protected void preInitChannel(final Channel channel) {
177    if (channel != null) {
178      final ChannelPipeline channelPipeline = channel.pipeline();
179      assert channelPipeline != null;
180      channelPipeline.addLast("LoggingHandler", new LoggingHandler());
181    }
182  }
183
184  /**
185   * A hook for performing {@link Channel} initialization after the
186   * Jersey integration is set up.
187   *
188   * <p>This implementation does nothing.</p>
189   *
190   * <p>Overrides must not call {@link #initChannel(Channel)} or an
191   * infinite loop will result.</p>
192   *
193   * @param channel the {@link Channel} being {@linkplain
194   * #initChannel(Channel) initialized}; may be {@code null} in which
195   * case no action will be taken
196   *
197   * @see #initChannel(Channel)
198   *
199   * @see ChannelPipeline#addLast(String, ChannelHandler)
200   */
201  protected void postInitChannel(final Channel channel) {
202
203  }
204
205  /**
206   * Creates and returns a new {@link SslHandler} when invoked.
207   *
208   * <p>This method never returns {@code null}.</p>
209   *
210   * <p>Overrides of this method must not return {@code null}.</p>
211   *
212   * @param sslContext the {@link SslContext} that may assist in the
213   * creation; must not be {@code null}
214   *
215   * @param byteBufAllocator a {@link ByteBufAllocator} that may
216   * assist in the creation; must not be {@code null}
217   *
218   * @return a new {@link SslHandler}; never {@code null}
219   *
220   * @see SslContext#newHandler(ByteBufAllocator)
221   */
222  protected SslHandler createSslHandler(final SslContext sslContext, final ByteBufAllocator byteBufAllocator) {
223    return sslContext.newHandler(byteBufAllocator);
224  }
