017package org.microbean.jersey.netty;
018
019import java.io.InputStream;
020import java.io.IOException;
021
022import java.util.Objects;
023
024import java.util.concurrent.Callable;
025import java.util.concurrent.ExecutionException;
026import java.util.concurrent.FutureTask;
027import java.util.concurrent.Phaser;
028
029import java.util.function.Function;
030
031import io.netty.buffer.ByteBuf;
032import io.netty.buffer.CompositeByteBuf;
033
034import io.netty.util.concurrent.EventExecutor;
035
036/**
037 * A {@link ByteBufQueue} and an {@link InputStream} that {@linkplain
038 * #read() reads} from a (possibly mutating and {@linkplain
039 * CompositeByteBuf#addComponent(boolean, ByteBuf) expanding}) {@link
040 * CompositeByteBuf}, ensuring that all operations occur only
041 * {@linkplain EventExecutor#inEventLoop() on the Netty event loop
042 * thread}.
043 *
044 * @author <a href="https://about.me/lairdnelson"
045 * target="_parent">Laird Nelson</a>
046 *
047 * @see #read(byte[], int, int)
048 *
049 * @see #read(Function)
050 */
051public class EventLoopPinnedByteBufInputStream extends InputStream implements ByteBufQueue {
052
053
054  /*
055   * Instance fields.
056   */
057
058  
059  private final CompositeByteBuf byteBuf;
060
061  private final EventExecutor eventExecutor;
062
063  private final Phaser phaser;
064
065
066  /*
067   * Constructors.
068   */
069
070  
071  /**
072   * Creates a new {@link EventLoopPinnedByteBufInputStream}.
073   *
074   * @param compositeByteBuf the {@link CompositeByteBuf} that will be
075   * {@linkplain CompositeByteBuf#readBytes(byte[], int, int) read}
076   * from; must not be {@code null}
077   *
078   * @param eventExecutor an {@link EventExecutor} that will
079   * {@linkplain EventExecutor#inEventLoop() ensure operations occur
080   * on the Netty event loop}; must not be {@code null}
081   *
082   * @exception NullPointerException if either {@code
083   * compositeByteBuf} or {@code eventExecutor} is {@code null}
084   */
085  public EventLoopPinnedByteBufInputStream(final CompositeByteBuf compositeByteBuf,
086                                           final EventExecutor eventExecutor) {
087    super();
088    this.phaser = new Phaser(2);
089    this.byteBuf = Objects.requireNonNull(compositeByteBuf);
090    this.eventExecutor = Objects.requireNonNull(eventExecutor);
091  }
092
093
094  /*
095   * InputStream overrides.
096   */
097
098
099  @Override
100  public final int read() throws IOException {
101    return this.read(sourceByteBuf -> Integer.valueOf(sourceByteBuf.readByte()));
102  }
103
104  @Override
105  public final int read(final byte[] targetBytes) throws IOException {
106    return this.read(targetBytes, 0, targetBytes.length);
107  }
108
109  @Override
110  public final int read(final byte[] targetBytes, final int offset, final int length) throws IOException {
111    Objects.requireNonNull(targetBytes);
112    final int returnValue;
113    if (offset < 0 || length < 0 || length > targetBytes.length - offset) {
114      throw new IndexOutOfBoundsException();
115    } else if (length == 0) {
116      returnValue = 0;
117    } else {
118      returnValue = this.read(sourceByteBuf -> {
119          final int readThisManyBytes = Math.min(length, sourceByteBuf.readableBytes());
120          sourceByteBuf.readBytes(targetBytes, offset, readThisManyBytes);
121          return Integer.valueOf(readThisManyBytes);
122        });
123    }
124    return returnValue;
125  }
126
127
128  /*
129   * ByteBufQueue methods.
130   */
131
132
133  /**
134   * {@linkplain CompositeByteBuf#addComponent(boolean, ByteBuf) Adds}
135   * the supplied {@link ByteBuf} to the {@link CompositeByteBuf}
136   * maintained by this {@link EventLoopPinnedByteBufInputStream}.
137   *
138   * @param byteBuf the {@link ByteBuf} to add; must not be {@code
139   * null}
140   *
141   * @exception NullPointerException if {@code byteBuf} is {@code
142   * null}
143   */
144  public final void addByteBuf(final ByteBuf byteBuf) {
145    Objects.requireNonNull(byteBuf);
146    if (byteBuf != this.byteBuf) {
147      if (this.eventExecutor.inEventLoop()) {
148        this.byteBuf.addComponent(true /* advance the writerIndex */, byteBuf);
149      } else {
150        this.eventExecutor.execute(() -> this.byteBuf.addComponent(true /* advance the writerIndex */, byteBuf));
151      }
152    }
153    this.phaser.arrive(); // (Nonblocking)
154  }
155
156
157  /*
158   * Protected methods.
159   */
160
161
162  /**
163   * {@linkplain Function#apply(Object) Applies} the supplied {@link
164   * Function} to a {@link ByteBuf}, ensuring that the {@link
165   * Function} application takes place on the {@linkplain
166   * EventExecutor#inEventLoop() Netty event loop thread}, and returns
167   * the result of invoking {@link Integer#intValue()} on the {@link
168   * Function}'s return value.
169   *
170   * @param function the {@link Function} to apply; must not be {@code
171   * null}; must not return {@code null}
172   *
173   * @return an {@code int} resulting from the {@link Function}
174   * application
175   *
176   * @exception NullPointerException if {@code function} is {@code null}
177   *
178   * @exception IOException if the return value of the supplied {@link
179   * Function} could not be acquired for some reason
180   */
181  protected final int read(final Function<? super ByteBuf, ? extends Integer> function) throws IOException {
182    Objects.requireNonNull(function);
183    final int phaseNumber = this.phaser.arrive();
184    while (this.byteBuf.numComponents() <= 0) {
185      this.phaser.awaitAdvance(phaseNumber); // BLOCKING
186    }
187    final FutureTask<Integer> readTask = new FutureTask<>(new EventLoopByteBufOperation(this.byteBuf, function));
188    if (this.eventExecutor.inEventLoop()) {
189      readTask.run();
190    } else {
191      this.eventExecutor.execute(readTask);
192    }
193    Integer returnValue = null;
194    try {
195      returnValue = readTask.get(); // BLOCKING
196    } catch (final ExecutionException executionException) {
197      final Throwable cause = executionException.getCause();
198      if (cause instanceof RuntimeException) {
199        throw (RuntimeException)cause;
200      } else if (cause instanceof Error) {
201        throw (Error)cause;
202      } else {
203        // This should be prevented by the compiler:
204        // EventLoopByteBufOperation#call() does not throw any checked
205        // exceptions.
206        throw new InternalError(cause.getMessage(), cause);
207      }
208    } catch (final InterruptedException interruptedException) {
209      Thread.currentThread().interrupt();
210      throw new IOException(interruptedException.getMessage(), interruptedException);
211    }
212    if (returnValue == null) {
213      throw new IOException("function.apply() == null");
214    }
215    return returnValue.intValue();
216  }
217
218
219  /*
220   * Inner and nested classes.
221   */
222
223  
224  private static final class EventLoopByteBufOperation implements Callable<Integer> {
225
226    private final ByteBuf byteBuf;
227
228    private final Function<? super ByteBuf, ? extends Integer> function;
229
230    private EventLoopByteBufOperation(final ByteBuf byteBuf,
231                                      final Function<? super ByteBuf, ? extends Integer> function) {
232      super();
233      this.byteBuf = Objects.requireNonNull(byteBuf);
234      this.function = Objects.requireNonNull(function);
235    }
236
237    @Override
238    public final Integer call() {
239      return this.byteBuf.refCnt() > 0 && this.byteBuf.isReadable() ? function.apply(this.byteBuf) : Integer.valueOf(-1);
240    }
241
242  }
