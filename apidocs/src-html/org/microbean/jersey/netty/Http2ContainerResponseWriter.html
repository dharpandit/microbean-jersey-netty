050/**
051 * An {@link AbstractNettyContainerResponseWriter} that works with
052 * {@link Http2HeadersFrame}-typed request objects.
053 *
054 *
055 * @author <a href="https://about.me/lairdnelson"
056 * target="_parent">Laird Nelson</a>
057 *
058 * @see AbstractNettyContainerResponseWriter
059 *
060 * @see HttpContainerResponseWriter
061 */
062public class Http2ContainerResponseWriter extends AbstractNettyContainerResponseWriter<Http2HeadersFrame> {
063
064
065  /*
066   * Static fields.
067   */
068
069
070  private static final String cn = Http2ContainerResponseWriter.class.getName();
071
072  private static final Logger logger = Logger.getLogger(cn);
073
074
075  /*
076   * Constructors.
077   */
078
079
080  /**
081   * Creates a new {@link Http2ContainerResponseWriter}.
082   *
083   * @param http2HeadersFrame the {@link Http2HeadersFrame} being
084   * responded to; must not be {@code null}
085   *
086   * @param channelHandlerContext the {@link ChannelHandlerContext}
087   * representing the current Netty execution; must not be {@code
088   * null}
089   *
090   * @param scheduledExecutorServiceSupplier a {@link Supplier} that
091   * can {@linkplain Supplier#get() supply} a {@link
092   * ScheduledExecutorService}; must not be {@code null}
093   *
094   * @exception NullPointerException if any of the parameters is
095   * {@code null}
096   */
097  public Http2ContainerResponseWriter(final Http2HeadersFrame http2HeadersFrame,
098                                      final ChannelHandlerContext channelHandlerContext,
099                                      final Supplier<? extends ScheduledExecutorService> scheduledExecutorServiceSupplier) {
100    super(http2HeadersFrame, channelHandlerContext, scheduledExecutorServiceSupplier);
101  }
102
103
104  /*
105   * Instance methods.
106   */
107
108
109  /**
110   * Implements the {@link
111   * AbstractNettyContainerResponseWriter#writeAndFlushStatusAndHeaders(ContainerResponse,
112   * long, ChannelPromise)} method by {@linkplain
113   * ChannelHandlerContext#writeAndFlush(Object) writing and flushing}
114   * an {@link DefaultHttp2HeadersFrame} object.
115   *
116   * @param containerResponse the {@link ContainerResponse} being
117   * processed; must not be {@code null}
118   *
119   * @param contentLength the length of the content in bytes; will be
120   * less than {@code 0} if the content length is unknown
121   *
122   * @exception NullPointerException if {@code containerResponse} or
123   * {@code channelPromise} is {@code null}
124   */
125  @Override
126  protected final void writeAndFlushStatusAndHeaders(final ContainerResponse containerResponse,
127                                                     final long contentLength,
128                                                     final ChannelPromise channelPromise) {
129    final String mn = "writeAndFlushStatusAndHeaders";
130    if (logger.isLoggable(Level.FINER)) {
131      logger.entering(cn, mn, new Object[] { containerResponse, Long.valueOf(contentLength) });
132    }
133    Objects.requireNonNull(containerResponse);
134    Objects.requireNonNull(channelPromise);
135
136    final Http2Headers nettyHeaders = new DefaultHttp2Headers();
137    nettyHeaders.status(Integer.toString(containerResponse.getStatus()));
138    copyHeaders(containerResponse.getStringHeaders(), String::toLowerCase, nettyHeaders::add);
139    if (contentLength >= 0L) {
140      // CONTENT_LENGTH is a constant that is guaranteed to be in
141      // lowercase so we aren't inconsistent.
142      nettyHeaders.set(HttpHeaderNames.CONTENT_LENGTH, Long.toString(contentLength));
143    }
144    this.channelHandlerContext.writeAndFlush(new DefaultHttp2HeadersFrame(nettyHeaders, contentLength == 0L), channelPromise);
145
146    if (logger.isLoggable(Level.FINER)) {
147      logger.exiting(cn, mn);
148    }
149  }
150
151  /**
152   * Implements the {@link
153   * AbstractNettyContainerResponseWriter#needsOutputStream(long)}
154   * method by returning {@code true} if and only if the supplied
155   * {@code contentLength} is not equal to {@code 0L} and if the
156   * {@linkplain Http2Headers#method() request method} {@linkplain
157   * String#toUpperCase() converted to uppercase} is not equal to the
158   * return value of invoking {@link HttpMethod#name()
159   * HttpMethod.HEAD.name()}.
160   *
161   * @param contentLength the length of the content in bytes; will be
162   * less than {@code 0} if the content length is unknown
163   *
164   * @return {@code true} if {@code contentLength} is not equal to
165   * {@code 0L} and if the {@linkplain Http2Headers#method() request
166   * method} {@linkplain String#toUpperCase() converted to uppercase}
167   * is not equal to the return value of invoking {@link
168   * HttpMethod#name() HttpMethod.HEAD.name()}; {@code false} in all
169   * other cases
170   */
171  @Override
172  protected final boolean needsOutputStream(final long contentLength) {
173    return contentLength != 0L && !HttpMethod.HEAD.equals(HttpMethod.valueOf(this.requestObject.headers().method().toString().toUpperCase()));
174  }
175
176  /**
177   * Returns a new {@link FunctionalByteBufChunkedInput
178   * FunctionalByteBufChunkedInput&lt;Http2DataFrame&gt;} when invoked.
179   *
180   * @param eventExecutor {@inheritDoc}
181   *
182   * @param source {@inheritDoc}
183   *
184   * @param contentLength {@inheritDoc}
185   *
186   * @return a new {@link FunctionalByteBufChunkedInput
187   * FunctionalByteBufChunkedInput&lt;Http2DataFrame&gt;}; never
188   * {@code null}
189   *
190   * @see AbstractNettyContainerResponseWriter#createChunkedInput(EventExecutor, ByteBuf, long)
191   *
192   * @see ChunkedInput#readChunk(ByteBufAllocator)
193   */
194  @Override
195  protected final BoundedChunkedInput<?> createChunkedInput(final EventExecutor eventExecutor, final ByteBuf source, final long contentLength) {
196    final String mn = "createChunkedInput";
197    if (logger.isLoggable(Level.FINER)) {
198      logger.entering(cn, mn, new Object[] { eventExecutor, source, Long.valueOf(contentLength) });
199    }
200    final BoundedChunkedInput<?> returnValue =  new FunctionalByteBufChunkedInput<Http2DataFrame>(source, DefaultHttp2DataFrame::new, contentLength);
201    if (logger.isLoggable(Level.FINER)) {
202      logger.exiting(cn, mn, returnValue);
203    }
204    return returnValue;
205  }
206
207  /**
208   * {@linkplain ChannelHandlerContext#write(Object) Writes} and
209   * implicitly flushes {@link
210   * DefaultHttp2DataFrame#DefaultHttp2DataFrame(boolean) new
211   * DefaultHttp2DataFrame(true)} when invoked.
212   *
213   * @param channelPromise a {@link ChannelPromise} to pass to any
214   * write operation; must not be {@code null}
215   *
216   * @exception NullPointerException if {@code channelPromise} is
217   * {@code null}
218   *
219   * @see DefaultHttp2DataFrame#DefaultHttp2DataFrame(boolean)
220   *
221   * @see
222   * AbstractNettyContainerResponseWriter#writeLastContentMessage(ChannelPromise)
223   */
224  @Override
225  protected final void writeLastContentMessage(final ChannelPromise channelPromise) {
226    Objects.requireNonNull(channelPromise);
227    this.channelHandlerContext.write(new DefaultHttp2DataFrame(true), channelPromise);
228  }
229
230  /**
231   * {@linkplain ChannelHandlerContext#write(Object) Writes} a new
232   * {@link DefaultHttp2Headers} message with a {@linkplain
233   * Http2Headers#status(CharSequence) status} equal to {@code 500}.
234   *
235   * @param channelPromise a {@link ChannelPromise} to pass to any
236   * write operation; must not be {@code null}
237   *
238   * @exception NullPointerException if {@code channelPromise} is
239   * {@code null}
240   *
241   * @see
242   * AbstractNettyContainerResponseWriter#writeFailureMessage(ChannelPromise)
243   */
244  @Override
245  protected final void writeFailureMessage(final ChannelPromise channelPromise) {
246    Objects.requireNonNull(channelPromise);
247    this.channelHandlerContext.write(new DefaultHttp2Headers().status(HttpResponseStatus.INTERNAL_SERVER_ERROR.codeAsText()),
248                                     channelPromise);
249  }
