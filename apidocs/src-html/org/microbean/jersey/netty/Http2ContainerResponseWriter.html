032
033import io.netty.handler.codec.http2.DefaultHttp2DataFrame;
034import io.netty.handler.codec.http2.DefaultHttp2Headers;
035import io.netty.handler.codec.http2.DefaultHttp2HeadersFrame;
036import io.netty.handler.codec.http2.Http2DataFrame;
037import io.netty.handler.codec.http2.Http2Headers;
038import io.netty.handler.codec.http2.Http2HeadersFrame;

046/**
047 * An {@link AbstractNettyContainerResponseWriter} that works with
048 * {@link Http2HeadersFrame}-typed request objects.
049 *
050 *
051 * @author <a href="https://about.me/lairdnelson"
052 * target="_parent">Laird Nelson</a>
053 *
054 * @see AbstractNettyContainerResponseWriter
055 *
056 * @see HttpContainerResponseWriter
057 */
058public class Http2ContainerResponseWriter extends AbstractNettyContainerResponseWriter<Http2HeadersFrame> {
059
060
061  /*
062   * Constructors.
063   */
064
065
066  /**
067   * Creates a new {@link Http2ContainerResponseWriter}.
068   *
069   * @param http2HeadersFrame the {@link Http2HeadersFrame} being
070   * responded to; must not be {@code null}
071   *
072   * @param channelHandlerContext the {@link ChannelHandlerContext}
073   * representing the current Netty execution; must not be {@code
074   * null}
075   *
076   * @param scheduledExecutorServiceSupplier a {@link Supplier} that
077   * can {@linkplain Supplier#get() supply} a {@link
078   * ScheduledExecutorService}; must not be {@code null}
079   *
080   * @exception NullPointerException if any of the parameters is
081   * {@code null}
082   */
083  public Http2ContainerResponseWriter(final Http2HeadersFrame http2HeadersFrame,
084                                      final ChannelHandlerContext channelHandlerContext,
085                                      final Supplier<? extends ScheduledExecutorService> scheduledExecutorServiceSupplier) {
086    super(http2HeadersFrame, channelHandlerContext, scheduledExecutorServiceSupplier);
087  }
088
089
090  /*
091   * Instance methods.
092   */
093
094
095  /**
096   * Implements the {@link
097   * AbstractNettyContainerResponseWriter#writeAndFlushStatusAndHeaders(ContainerResponse,
098   * long)} method by {@linkplain
099   * ChannelHandlerContext#writeAndFlush(Object) writing and flushing}
100   * an {@link DefaultHttp2HeadersFrame} object.
101   *
102   * @param containerResponse the {@link ContainerResponse} being
103   * processed; must not be {@code null}
104   *
105   * @param contentLength the length of the content in bytes; will be
106   * less than {@code 0} if the content length is unknown
107   *
108   * @exception NullPointerException if {@code containerResponse} is
109   * {@code null}
110   */
111  @Override
112  protected final void writeAndFlushStatusAndHeaders(final ContainerResponse containerResponse,
113                                                     final long contentLength) {
114    Objects.requireNonNull(containerResponse);
115
116    final Http2Headers nettyHeaders = new DefaultHttp2Headers();
117    nettyHeaders.status(Integer.toString(containerResponse.getStatus()));
118    copyHeaders(containerResponse.getStringHeaders(), String::toLowerCase, nettyHeaders::add);
119    if (contentLength >= 0L) {
120      // CONTENT_LENGTH is a constant that is guaranteed to be in
121      // lowercase so we aren't inconsistent.
122      nettyHeaders.set(HttpHeaderNames.CONTENT_LENGTH, Long.toString(contentLength));
123    }
124    this.channelHandlerContext.writeAndFlush(new DefaultHttp2HeadersFrame(nettyHeaders, contentLength == 0L));
125  }
126
127  /**
128   * Implements the {@link
129   * AbstractNettyContainerResponseWriter#needsOutputStream(long)}
130   * method by returning {@code true} if and only if the supplied
131   * {@code contentLength} is not equal to {@code 0L} and if the
132   * {@linkplain Http2Headers#method() request method} {@linkplain
133   * String#toUpperCase() converted to uppercase} is not equal to the
134   * return value of invoking {@link HttpMethod#name()
135   * HttpMethod.HEAD.name()}.
136   *
137   * @param contentLength the length of the content in bytes; will be
138   * less than {@code 0} if the content length is unknown
139   *
140   * @return {@code true} if {@code contentLength} is not equal to
141   * {@code 0L} and if the {@linkplain Http2Headers#method() request
142   * method} {@linkplain String#toUpperCase() converted to uppercase}
143   * is not equal to the return value of invoking {@link
144   * HttpMethod#name() HttpMethod.HEAD.name()}; {@code false} in all
145   * other cases
146   */
147  @Override
148  protected final boolean needsOutputStream(final long contentLength) {
149    return contentLength != 0L && !HttpMethod.HEAD.equals(HttpMethod.valueOf(this.requestObject.headers().method().toString().toUpperCase()));
150  }
151
152  /**
153   * Returns a new {@link FunctionalByteBufChunkedInput
154   * FunctionalByteBufChunkedInput&lt;Http2DataFrame&gt;} when invoked.
155   *
156   * @param eventExecutor {@inheritDoc}
157   *
158   * @param source {@inheritDoc}
159   *
160   * @param contentLength {@inheritDoc}
161   *
162   * @return a new {@link FunctionalByteBufChunkedInput
163   * FunctionalByteBufChunkedInput&lt;Http2DataFrame&gt;}; never
164   * {@code null}
165   *
166   * @see AbstractNettyContainerResponseWriter#createChunkedInput(EventExecutor, ByteBuf, long)
167   *
168   * @see ChunkedInput#readChunk(ByteBufAllocator)
169   */
170  @Override
171  protected final ChunkedInput<?> createChunkedInput(final EventExecutor eventExecutor, final ByteBuf source, final long contentLength) {
172    //return new FunctionalByteBufChunkedInput<Http2DataFrame>(source, bb -> new DefaultHttp2DataFrame(bb), contentLength);
173    return new FunctionalByteBufChunkedInput<Http2DataFrame>(source, DefaultHttp2DataFrame::new, contentLength);
174  }
175
176  /**
177   * {@linkplain ChannelHandlerContext#writeAndFlush(Object) Writes
178   * and flushes} {@link
179   * DefaultHttp2DataFrame#DefaultHttp2DataFrame(boolean) new
180   * DefaultHttp2DataFrame(true)} when invoked.
181   *
182   * @see DefaultHttp2DataFrame#DefaultHttp2DataFrame(boolean)
183   *
184   * @see AbstractNettyContainerResponseWriter#writeLastContentMessage()
185   */
186  @Override
187  protected final void writeLastContentMessage() {
188    this.channelHandlerContext.write(new DefaultHttp2DataFrame(true));
189  }
190
191  /**
192   * {@linkplain ChannelHandlerContext#write(Object) Writes} a new
193   * {@link DefaultHttp2Headers} message with a {@linkplain
194   * Http2Headers#status(CharSequence) status} equal to {@code 500}.
195   *
196   * @see AbstractNettyContainerResponseWriter#writeFailureMessage()
197   */
198  @Override
199  protected final void writeFailureMessage() {
200    this.channelHandlerContext.write(new DefaultHttp2Headers().status(HttpResponseStatus.INTERNAL_SERVER_ERROR.codeAsText()));
201  }
202
203}
