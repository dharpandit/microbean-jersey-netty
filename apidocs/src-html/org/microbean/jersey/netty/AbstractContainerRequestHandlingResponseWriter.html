053/**
054 * An abstract {@link ChannelInboundHandlerAdapter} that is also a
055 * {@link ContainerResponseWriter} that processes incoming {@link
056 * ContainerRequest} events, such as those dispatched by an {@link
057 * AbstractContainerRequestDecoder}, by supplying them to the {@link
058 * ApplicationHandler#handle(ContainerRequest)} method.
059 *
060 * <p>Instances of this class are in charge of properly invoking
061 * {@link ApplicationHandler#handle(ContainerRequest)}, thus adapting
062 * <a href="https://eclipse-ee4j.github.io/jersey/"
063 * target="_parent">Jersey</a> to <a href="https://netty.io/"
064 * target="_parent">Netty</a>'s constraints and vice versa.</p>
065 *
066 * @param <T> the type of message that will be written by instances of
067 * this class; see {@link #createOutputStream(long,
068 * ContainerResponse)}
069 *
070 * @author <a href="https://about.me/lairdnelson"
071 * target="_parent">Laird Nelson</a>
072 *
073 * @see #channelRead(ChannelHandlerContext, Object)
074 *
075 * @see #createOutputStream(long, ContainerResponse)
076 * 
077 * @see ChannelInboundHandlerAdapter
078 *
079 * @see ApplicationHandler#handle(ContainerRequest)
080 *
081 * @see ContainerResponseWriter
082 */
083public abstract class AbstractContainerRequestHandlingResponseWriter<T> extends ChannelInboundHandlerAdapter implements ContainerResponseWriter {

217  /**
218   * If the supplied {@code message} is a {@link ContainerRequest}
219   * then this method will {@linkplain
220   * ContainerRequest#setWriter(ContainerResponseWriter) install
221   * itself as that request's <code>ContainerResponseWriter</code>}
222   * and will invoke {@link
223   * ApplicationHandler#handle(ContainerRequest)}.
224   *
225   * <p>In all other cases this method will simply call {@link
226   * ChannelInboundHandlerAdapter#channelRead(ChannelHandlerContext,
227   * Object)} with the supplied {@code message}.</p>
228   *
229   * @param channelHandlerContext the {@link ChannelHandlerContext} in
230   * effect; must not be {@code null}
231   *
232   * @param message the incoming message, or event; may be {@code null}
233   *
234   * @exception NullPointerException if {@code channelHandlerContext}
235   * is {@code null}
236   *
237   * @exception Exception if {@code message} is not an instance of
238   * {@link ContainerRequest} and {@link
239   * ChannelInboundHandlerAdapter#channelRead(ChannelHandlerContext,
240   * Object)} throws an {@link Exception}
241   *
242   * @see ApplicationHandler#handle(ContainerRequest)
243   *
244   * @see ContainerResponseWriter
245   *
246   * @see ContainerRequest#setWriter(ContainerResponseWriter)
247   */
248  @Override
249  public final void channelRead(final ChannelHandlerContext channelHandlerContext,
250                                final Object message)
251    throws Exception {
252    if (this.getChannelHandlerContext() != null) {
253      throw new IllegalStateException("this.getChannelHandlerContext() != null: " + this.getChannelHandlerContext());
254    }
255    this.channelHandlerContext = Objects.requireNonNull(channelHandlerContext);
256    try {
257      if (message instanceof ContainerRequest) {
258        final ContainerRequest containerRequest = (ContainerRequest)message;
259        containerRequest.setWriter(this);
260        this.applicationHandler.handle(containerRequest);
261      } else {
262        super.channelRead(channelHandlerContext, message);
263      }
264    } finally {
265      this.channelHandlerContext = null;
266    }
267  }

269  /**
270   * Overrides the {@link
271   * ChannelInboundHandlerAdapter#channelReadComplete(ChannelHandlerContext)}
272   * method to call {@link ChannelHandlerContext#flush()
273   * channelHandlerContext.flush()} before calling the superclass
274   * implementation.
275   *
276   * @param channelHandlerContext the {@link ChannelHandlerContext} in
277   * effect; must not be {@code null}
278   *
279   * @exception NullPointerException if {@code channelHandlerContext} is {@code null}
280   *
281   * @exception Exception if {@link
282   * ChannelInboundHandlerAdapter#channelReadComplete(ChannelHandlerContext)}
283   * throws an {@link Exception}
284   */
285  @Override
286  public final void channelReadComplete(final ChannelHandlerContext channelHandlerContext) throws Exception {
287    channelHandlerContext.flush();
288    // See
289    // https://github.com/netty/netty/blob/d446765b8469ca40db40f46e5c637d980b734a8a/transport/src/main/java/io/netty/channel/DefaultChannelPipeline.java#L1408-L1413.
290    super.channelReadComplete(channelHandlerContext);
291    if (!channelHandlerContext.channel().config().isAutoRead()) {
292      // Ultimately a read is just an idempotent call so even if other
293      // handlers do this seemingly nothing bad will happen.  See
294      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport/src/main/java/io/netty/channel/nio/AbstractNioChannel.java#L402-L416.
295      // Even in the ancient case of blocking IO it's still
296      // idempotent:
297      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport/src/main/java/io/netty/channel/oio/AbstractOioChannel.java#L101-L109
298      // Finally, transports like Epoll are less clear:
299      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport-native-epoll/src/main/java/io/netty/channel/epoll/AbstractEpollChannel.java#L226-L242
300      // ...but Epoll too ultimately is idempotent:
301      // https://github.com/netty/netty/blob/9976ab7fe86e052d29ca7accf528c885e93dcb4c/transport-native-epoll/src/main/java/io/netty/channel/epoll/AbstractEpollChannel.java#L226-L242
302      channelHandlerContext.read();
303    }
304  }

402   *

406  public final OutputStream writeResponseStatusAndHeaders(final long contentLength,
407                                                          final ContainerResponse containerResponse) {
408    final OutputStream returnValue;
409    if (this.writeStatusAndHeaders(contentLength, Objects.requireNonNull(containerResponse)) && contentLength != 0L) {

411    } else {
412      returnValue = null;
413    }
414    return returnValue;

574  @Override
575  public final boolean suspend(final long timeout,
576                               final TimeUnit timeUnit,
577                               final TimeoutHandler timeoutHandler) {
578    // Lifted from Jersey's supplied Netty integration, with repairs.
579    final boolean returnValue;
580    if (timeoutHandler == null || this.suspendTimeoutHandler != null) {
581      returnValue = false;
582    } else {
583      this.suspendTimeoutHandler = () -> {
584        timeoutHandler.onTimeout(this);
585        this.suspendTimeoutHandler = null;
586      };
587      if (timeout > 0L) {
588        final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
589        this.suspendTimeoutFuture =
590          channelHandlerContext.executor().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
591      }
592      returnValue = true;
593    }
594    return returnValue;
595  }
596
597  @Override
598  public final void setSuspendTimeout(final long timeout, final TimeUnit timeUnit) {
599    // Lifted from Jersey's supplied Netty integration, with repairs.
600    if (this.suspendTimeoutHandler == null) {
601      throw new IllegalStateException("this.suspendTimeoutHandler == null");
602    }
603    if (this.suspendTimeoutFuture != null) {
604      this.suspendTimeoutFuture.cancel(true);
605      this.suspendTimeoutFuture = null;
606    }
607    if (timeout > 0L) {
608      final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
609      this.suspendTimeoutFuture =
610        channelHandlerContext.executor().schedule(this.suspendTimeoutHandler, timeout, timeUnit);
611    }
612  }

614  /**
615   * Handles any failure case encountered by the logic encapsulated by
616   * the {@link ApplicationHandler#handle(ContainerRequest)} method.
617   *
618   * <p>Either {@link #commit()} or {@link #failure(Throwable)} will
619   * be called by Jersey as the logically final operation in the logic
620   * encapsulated by the {@link
621   * ApplicationHandler#handle(ContainerRequest)} method, but not
622   * both.</p>
623   *
624   * <p>This method calls the {@link #writeFailureMessage(Throwable)}
625   * method and takes great care to ensure that any {@link Throwable}s
626   * encountered along the way are properly recorded and {@linkplain
627   * Throwable#addSuppressed(Throwable) suppressed}.</p>
628   *
629   * <p><strong>This implementation never returns.</strong> A {@link
630   * ContainerException} is always thrown by this method.</p>
631   *
632   * @param failureCause the {@link Throwable} encountered by the
633   * {@link ApplicationHandler#handle(ContainerRequest)} method; may
634   * be {@code null}
635   *
636   * @exception ContainerException when this method is invoked; it
637   * will have the supplied {@code failureCause} as its {@linkplain
638   * Throwable#getCause() cause}
639   *
640   * @see #commit()
641   *
642   * @see ContainerResponseWriter#failure(Throwable)
643   */
644  @Override
645  public final void failure(final Throwable failureCause) {
646    final ChannelHandlerContext channelHandlerContext = Objects.requireNonNull(this.getChannelHandlerContext());
647    Throwable outerWriteProblem = null;
648    try {
649      this.writeFailureMessage(failureCause);
650    } catch (final RuntimeException | Error writeProblem) {
651      outerWriteProblem = writeProblem;
652      if (failureCause != null) {
653        boolean foundFailureCauseInSuppressedThrowables = false;
654        final Object[] suppressedThrowables = writeProblem.getSuppressed();
655        if (suppressedThrowables != null && suppressedThrowables.length > 0) {
656          for (final Object suppressedThrowable : suppressedThrowables) {
657            if (suppressedThrowable == failureCause) {
658              foundFailureCauseInSuppressedThrowables = true;
659              break;
660            }
661          }
662        }
663        if (!foundFailureCauseInSuppressedThrowables) {
664          writeProblem.addSuppressed(failureCause);
665        }
666      }
667      throw writeProblem;
668    } finally {
669      Throwable outerFlushProblem = null;
670      try {
671        channelHandlerContext.flush();
672      } catch (final RuntimeException | Error flushProblem) {
673        outerFlushProblem = flushProblem;
674        if (outerWriteProblem != null) {
675          flushProblem.addSuppressed(outerWriteProblem);
676        } else if (failureCause != null) {
677          flushProblem.addSuppressed(failureCause);
678        }
679        throw flushProblem;
680      } finally {
681        try {
682          channelHandlerContext.close();
683        } catch (final RuntimeException | Error closeProblem) {
684          if (outerFlushProblem != null) {
685            closeProblem.addSuppressed(outerFlushProblem);
686          } else if (failureCause != null) {
687            closeProblem.addSuppressed(failureCause);
688          }
689          throw closeProblem;
690        }
691      }
692    }
693    if (failureCause == null) {
694      throw new ContainerException("failure");
695    } else if (failureCause instanceof RuntimeException) {
696      throw (RuntimeException)failureCause;
697    } else if (failureCause instanceof Exception) {
698      throw new ContainerException(failureCause.getMessage(), failureCause);
699    } else {
700      throw (Error)failureCause;
701    }
702  }

705   * Writes an appropriate message, possibly using the {@link
706   * #getChannelHandlerContext() ChannelHandlerContext} to do so.
707   *
708   * <p>Implementations of this method must not call the {@link
709   * #failure(Throwable)} method or an infinite loop may result.</p>
710   *
711   * @param failureCause the {@link Throwable} responsible for this
712   * method's invocation; may be {@code null} in pathological cases
713   */
714  protected abstract void writeFailureMessage(final Throwable failureCause);

722  /**
723   * A utility function that copies entries from a source {@link Map}
724   * by passing each entry to the supplied {@link BiConsumer},
725   * transforming the keys beforehand using the supplied {@link
726   * UnaryOperator} and that is intended in this framework to be used
727   * to copy HTTP or HTTP/2 headers to and from the proper places.
728   *
729   * @param headersSource the source of the headers to copy; may be
730   * {@code null} in which case no action will be taken
731   *
732   * @param keyTransformer a {@link UnaryOperator} that transforms a
733   * header name; if {@code null} then the return value of {@link
734   * UnaryOperator#identity()} will be used instead
735   *
736   * @param headersTarget where the headers will be copied to; may be
737   * {@code null} in which case no action will be taken
738   */
739  protected static final void copyHeaders(final Map<? extends String, ? extends List<String>> headersSource,
740                                          UnaryOperator<String> keyTransformer,
741                                          final BiConsumer<? super String, ? super List<String>> headersTarget) {
742    if (headersTarget != null && headersSource != null && !headersSource.isEmpty()) {
743      final Collection<? extends Entry<? extends String, ? extends List<String>>> entrySet = headersSource.entrySet();
744      if (entrySet != null && !entrySet.isEmpty()) {
745        if (keyTransformer == null) {
746          keyTransformer = UnaryOperator.identity();
747        }
748        for (final Entry<? extends String, ? extends List<String>> entry : entrySet) {
749          if (entry != null) {
750            headersTarget.accept(keyTransformer.apply(entry.getKey()), entry.getValue());
751          }
752        }
753      }
754    }
755  }
