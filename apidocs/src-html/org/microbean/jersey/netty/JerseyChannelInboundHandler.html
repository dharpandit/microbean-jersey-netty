067/**
068 * A {@link SimpleChannelInboundHandler} that adapts <a
069 * href="https://jersey.github.io/" target="_parent">Jersey</a> to <a
070 * href="https://netty.io/" target="_parent">Netty</a>.
071 *
072 * <h2>Thread Safety</h2>
073 *
074 * <p>Instances of this class are safe for concurrent use by multiple
075 * threads, but certain methods must be invoked {@linkplain
076 * EventExecutor#inEventLoop() in the Netty event loop}.</p>
077 *
078 * @author <a href="https://about.me/lairdnelson"
079 * target="_parent">Laird Nelson</a>
080 *
081 * @see JerseyChannelInitializer
082 */

220  /**
221   * Creates a new {@link JerseyChannelInboundHandler}.
222   *
223   * @param baseUri the base {@link URI} for the Jersey application;
224   * may be {@code null} in which case the return value resulting from
225   * invoking {@link URI#create(String) URI.create("/")} will be used
226   * instead
227   *
228   * @param applicationHandler the Jersey {@link ApplicationHandler}
229   * that will actually run the Jersey application; may be {@code
230   * null} in which case a {@linkplain
231   * ApplicationHandler#ApplicationHandler() new} {@link
232   * ApplicationHandler} will be used instead
233   *
234   * @param jerseyExecutor the {@link Executor} that will be used to
235   * {@linkplain Executor#execute(Runnable) execute} a Jersey
236   * application; may be {@code null} in which case the return value
237   * of {@link ExecutorServiceProvider#getExecutorService()
238   * applicationHandler.getInjectionManager().getInstance(ExecutorServiceProvider.class).getExecutorService()}
239   * will be used instead
240   *
241   * @param jerseyScheduledExecutorServiceSupplier a {@link Supplier}
242   * of {@link ScheduledExecutorService} instances that may be used by
243   * {@link ContainerResponseWriter} instances involved in this {@link
244   * JerseyChannelInboundHandler}; may be {@code null} in which case
245   * the return value of {@linkplain
246   * ScheduledExecutorServiceProvider#getExecutorService()
247   * applicationHandler.getInjectionManager().getInstance(ScheduledExecutorServiceProvider.class).getExecutorService()}
248   * will be used instead
249   *
250   * @exception IllegalStateException if either {@code jerseyExecutor}
251   * or {@code jerseyScheduledExecutorServiceSupplier} is {@code null}
252   * and {@link ApplicationHandler#getInjectionManager()
253   * applicationHandler.getInjectionManager()} returns {@code null}
254   *
255   * @exception NullPointerException if {@link
256   * ExecutorServiceProvider#getExecutorService()
257   * applicationHandler.getInjectionManager().getInstance(ExecutorServiceProvider.class).getExecutorService()}
258   * returns {@code null}
259   *
260   * @see ApplicationHandler#getInjectionManager()
261   *
262   * @see #createContainerResponseWriter(HttpRequest,
263   * ChannelHandlerContext, Supplier)
264   *
265   * @see #createContainerResponseWriter(Http2HeadersFrame,
266   * ChannelHandlerContext, Supplier)
267   */
268  public JerseyChannelInboundHandler(final URI baseUri,
269                                     final ApplicationHandler applicationHandler,
270                                     Executor jerseyExecutor,
271                                     Supplier<? extends ScheduledExecutorService> jerseyScheduledExecutorServiceSupplier) {
272    super();
273    this.baseUri = baseUri == null ? URI.create("/") : baseUri;
274    this.applicationHandler = applicationHandler == null ? new ApplicationHandler() : applicationHandler;
275    InjectionManager injectionManager = null;
276    if (jerseyExecutor == null) {
277      injectionManager = this.applicationHandler.getInjectionManager();
278      if (injectionManager == null) {
279        throw new IllegalStateException("applicationHandler.getInjectionManager() == null");
280      }
281      jerseyExecutor = Objects.requireNonNull(injectionManager.getInstance(ExecutorServiceProvider.class).getExecutorService());
282    }
283    this.jerseyExecutor = jerseyExecutor;
284    if (jerseyScheduledExecutorServiceSupplier == null) {
285      if (injectionManager == null) {
286        injectionManager = this.applicationHandler.getInjectionManager();
287        if (injectionManager == null) {
288          throw new IllegalStateException("applicationHandler.getInjectionManager() == null");
289        }
290      }
291      final ScheduledExecutorServiceProvider provider = Objects.requireNonNull(injectionManager.getInstance(ScheduledExecutorServiceProvider.class));
292      jerseyScheduledExecutorServiceSupplier = () -> provider.getExecutorService();
293    }
294    this.jerseyScheduledExecutorServiceSupplier = jerseyScheduledExecutorServiceSupplier;
295  }

325  /**
326   * Arranges for the {@link ApplicationHandler} {@linkplain
327   * #JerseyChannelInboundHandler(URI, ApplicationHandler)
328   * supplied at construction time} to process the HTTP or HTTP/2
329   * message represented by the supplied {@link Object}.
330   *
331   * <p>This method must be invoked {@linkplain
332   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
333   *
334   * @param channelHandlerContext the {@link ChannelHandlerContext}
335   * representing the current Netty execution; must not be {@code
336   * null}
337   *
338   * @param message the {@link ByteBufHolder}&mdash;an {@link
339   * HttpRequest} or a {@link HttpContent}&mdash;to process; must not
340   * be {@code null}
341   *
342   * @exception IllegalArgumentException if {@code message} is not an
343   * instance of any of the following types: {@link HttpRequest},
344   * {@link Http2HeadersFrame}, {@link HttpContent} or {@link
345   * Http2DataFrame}
346   *
347   * @exception IllegalStateException if any methods of this class
348   * have been overridden incorrectly, or in general if any
349   * preconditions have been violated
350   *
351   * @see #messageReceived(ChannelHandlerContext, ByteBuf, boolean)
352   */
353  @Override
354  protected final void channelRead0(final ChannelHandlerContext channelHandlerContext,
355                                    final Object message) {
356    final String mn = "channelRead0";
357    if (logger.isLoggable(Level.FINER)) {
358      logger.entering(cn, mn, new Object[] { channelHandlerContext, message });
359    }
360
361    Objects.requireNonNull(channelHandlerContext);
362    if (message instanceof HttpRequest || message instanceof Http2HeadersFrame) {
363      this.messageReceived(channelHandlerContext, message);
364    } else if (message instanceof HttpContent || message instanceof Http2DataFrame) {
365      this.messageReceived(channelHandlerContext,
366                           ((ByteBufHolder)message).content(),
367                           message instanceof LastHttpContent || ((message instanceof Http2DataFrame) && ((Http2DataFrame)message).isEndStream()));
368    } else {
369      throw new IllegalArgumentException("Unexpected message type: " + message);
370    }
371
372    if (logger.isLoggable(Level.FINER)) {
373      logger.exiting(cn, mn);
374    }
375  }

377  /**
378   * Called by the {@link #channelRead0(ChannelHandlerContext,
379   * Object)} method when either an {@link HttpRequest} or an {@link
380   * Http2HeadersFrame} is received.
381   *
382   * <p>This method calls the {@link
383   * ApplicationHandler#handle(ContainerRequest)} method on a separate
384   * thread to start the Jersey application handling process.</p>
385   *
386   * @param channelHandlerContext the {@link ChannelHandlerContext}
387   * representing the current Netty execution; must not be {@code
388   * null}
389   *
390   * @param requestObject either an {@link HttpRequest} or an {@link
391   * Http2HeadersFrame}; must not be {@code null}
392   *
393   * @exception NullPointerException if {@code channelHandlerContext}
394   * or {@code requestObject} is {@code null}
395   *
396   * @exception IllegalArgumentException if somehow {@code
397   * requestObject} is neither an {@link HttpRequest} nor an {@link
398   * Http2HeadersFrame} (this should be impossible)
399   *
400   * @exception IllegalStateException if an override of any of the
401   * {@link #createContainerRequest(ChannelHandlerContext,
402   * HttpRequest)}, {@link
403   * #createContainerRequest(ChannelHandlerContext,
404   * Http2HeadersFrame)}, {@link
405   * #createContainerResponseWriter(HttpRequest,
406   * ChannelHandlerContext, Supplier)}, or {@link
407   * #createContainerResponseWriter(Http2HeadersFrame,
408   * ChannelHandlerContext, Supplier)} methods returns {@code null}
409   *
410   * @see <a
411   * href="https://github.com/microbean/microbean-jersey-netty/issues/8"
412   * target="_parent">Issue #8</a>
413   */
414  private final void messageReceived(final ChannelHandlerContext channelHandlerContext,
415                                     final Object requestObject) {
416    final String mn = "messageReceived";
417    if (logger.isLoggable(Level.FINER)) {
418      logger.entering(cn, mn, new Object[] { channelHandlerContext, requestObject });
419    }
420
421    Objects.requireNonNull(channelHandlerContext);
422    Objects.requireNonNull(requestObject);
423    assert channelHandlerContext.executor().inEventLoop();
424    assert this.byteBufQueue == null;
425
426    final ContainerRequest containerRequest;
427    final ContainerResponseWriter writer;
428    if (requestObject instanceof HttpRequest) {
429      final HttpRequest httpRequest = (HttpRequest)requestObject;
430      containerRequest = this.createContainerRequest(channelHandlerContext, httpRequest);
431      writer = this.createContainerResponseWriter(httpRequest,
432                                                  channelHandlerContext,
433                                                  this.jerseyScheduledExecutorServiceSupplier);
434    } else if (requestObject instanceof Http2HeadersFrame) {
435      final Http2HeadersFrame http2HeadersFrame = (Http2HeadersFrame)requestObject;
436      containerRequest = this.createContainerRequest(channelHandlerContext, http2HeadersFrame);
437      writer = this.createContainerResponseWriter(http2HeadersFrame,
438                                                  channelHandlerContext,
439                                                  this.jerseyScheduledExecutorServiceSupplier);
440    } else {
441      throw new IllegalArgumentException("Unexpected requestObject: " + requestObject);
442    }
443    if (containerRequest == null) {
444      throw new IllegalStateException("createContainerRequest() == null");
445    }
446    if (writer == null) {
447      throw new IllegalStateException("createContainerResponseWriter() == null");
448    }
449    containerRequest.setWriter(writer);
450
451    this.jerseyExecutor.execute(() -> {
452        try {
453          this.applicationHandler.handle(containerRequest);
454        } catch (final RuntimeException | Error problem) {
455          // If Jersey is well-behaved, this will never happen,
456          // because the containerResponseWriter's failure(Throwable)
457          // method will have been called.
458          if (logger.isLoggable(Level.SEVERE)) {
459            logger.logp(Level.SEVERE, cn, mn, problem.getMessage(), problem);
460          }
461          // With ordinary ExecutorService implementations, this goes
462          // to /dev/null effectively.
463          throw problem;
464        }
465      });
466
467    if (logger.isLoggable(Level.FINER)) {
468      logger.exiting(cn, mn);
469    }
470  }

472  /**
473   * Processes the supplied incoming {@link ByteBuf} representing one
474   * of possibly many "content" portions of an overall HTTP message.
475   *
476   * <p>Internally, the supplied {@code content}, if {@linkplain
477   * ByteBuf#isReadable() readable}, is {@linkplain ByteBuf#retain()
478   * retained} and {@linkplain ByteBufQueue#addByteBuf(ByteBuf) added
479   * to an internal <code>ByteBufQueue</code> implementation} created
480   * by the {@link #channelRead0(ChannelHandlerContext, Object)}
481   * method.
482   *
483   * <p>This method will be invoked {@linkplain
484   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
485   *
486   * @param channelHandlerContext the {@link ChannelHandlerContext}
487   * representing the current Netty execution; may be {@code
488   * null}
489   *
490   * @param content the {@link ByteBuf} to process; may be {@code
491   * null} somewhat pathologically
492   *
493   * @param lastOne {@code true} if {@code content} is known to be the
494   * last such content chunk; {@code false} in all other cases
495   *
496   * @see #channelRead0(ChannelHandlerContext, Object)
497   */
498  private final void messageReceived(final ChannelHandlerContext channelHandlerContext,
499                                     final ByteBuf content,
500                                     final boolean lastOne) {
501    final String mn = "messageReceived";
502    if (logger.isLoggable(Level.FINER)) {
503      logger.entering(cn, mn, new Object[] { channelHandlerContext, content, Boolean.valueOf(lastOne) });
504    }
505
506    final ByteBufQueue byteBufQueue = this.byteBufQueue;
507
508    // Handle incoming payloads.
509    if (content != null && content.isReadable()) {
510      assert byteBufQueue != null;
511      // TODO: do we actually need to retain this, given that it's
512      // going to end up in a CompositeByteBuf?
513      content.retain();
514      byteBufQueue.addByteBuf(content);
515    }
516
517    // If that was the last incoming payload chunk, then we have no
518    // more need for our ByteBufQueue: we've added all we're ever
519    // going to add.  This will help to fail fast if somehow more
520    // content comes in.
521    if (lastOne && byteBufQueue != null) {
522      this.byteBufQueue = null;
523    }
524
525    if (logger.isLoggable(Level.FINER)) {
526      logger.exiting(cn, mn);
527    }
528  }

602  /**
603   * A method to which both the {@link
604   * #createContainerRequest(ChannelHandlerContext,
605   * Http2HeadersFrame)} and {@link
606   * #createContainerRequest(ChannelHandlerContext, HttpRequest)}
607   * delegate.
608   *
609   * @param channelHandlerContext the {@link ChannelHandlerContext}
610   * representing the current Netty execution; must not be {@code
611   * null}
612   *
613   * @param requestObject the {@link HttpRequest} or {@link
614   * Http2HeadersFrame} from which a {@link ContainerRequest} should
615   * be synthesized
616   *
617   * @return a new {@link ContainerRequest}; never {@code null}
618   *
619   * @exception NullPointerException if either {@code
620   * channelHandlerContext} or {@code requestObject} is {@code null}
621   *
622   * @exception IllegalArgumentException if somehow {@code
623   * requestObject} is neither an {@link HttpRequest} nor an {@link
624   * Http2HeadersFrame} (this should be impossible)
625   *
626   * @see ContainerRequest
627   *
628   * @see #createContainerRequest(ChannelHandlerContext,
629   * Http2HeadersFrame)
630   *
631   * @see #createContainerRequest(ChannelHandlerContext,
632   * HttpRequest) 
633   */
634  private final ContainerRequest createContainerRequest(final ChannelHandlerContext channelHandlerContext,
635                                                        final Object requestObject) {
636    final String mn = "createContainerRequest";
637    if (logger.isLoggable(Level.FINER)) {
638      logger.entering(cn, mn, new Object[] { channelHandlerContext, requestObject });
639    }
640
641    Objects.requireNonNull(channelHandlerContext);
642    Objects.requireNonNull(requestObject);
643    assert channelHandlerContext.executor().inEventLoop();
644
645    final String method;
646    final String uriString;
647    final Iterable<? extends CharSequence> nettyHeaderNames;
648    if (requestObject instanceof HttpRequest) {
649      final HttpRequest httpRequest = (HttpRequest)requestObject;
650      final HttpHeaders httpHeaders = httpRequest.headers();
651      assert httpHeaders != null;
652      nettyHeaderNames = httpHeaders.names();
653      method = httpRequest.method().name();
654      uriString = httpRequest.uri();
655    } else if (requestObject instanceof Http2HeadersFrame) {
656      final Http2HeadersFrame http2HeadersFrame = (Http2HeadersFrame)requestObject;
657      final Http2Headers headers = http2HeadersFrame.headers();
658      nettyHeaderNames = headers.names();
659      method = headers.method().toString();
660      uriString = headers.path().toString();
661    } else {
662      throw new IllegalArgumentException("Unexpected requestObject: " + requestObject);
663    }
664    assert method != null;
665    assert uriString != null;
666
667    final SecurityContext securityContext = this.createSecurityContext(channelHandlerContext, requestObject);
668
669    final ContainerRequest returnValue =
670      new ContainerRequest(this.baseUri,
671                           baseUri.resolve(ContainerUtils.encodeUnsafeCharacters(uriString.startsWith("/") && uriString.length() > 1 ? uriString.substring(1) : uriString)),
672                           method,
673                           securityContext == null ? new SecurityContextAdapter() : securityContext,
674                           new MapBackedPropertiesDelegate());
675
676    final BiConsumer<? super ContainerRequest, ? super CharSequence> headersInstaller;
677    if (requestObject instanceof HttpRequest) {
678      headersInstaller = (containerRequest, name) -> containerRequest.headers(name.toString(), ((HttpRequest)requestObject).headers().getAll(name));
679    } else {
680      assert requestObject instanceof Http2HeadersFrame;
681      headersInstaller = (containerRequest, name) -> containerRequest.headers(name.toString(), (Iterable<CharSequence>)() -> ((Http2HeadersFrame)requestObject).headers().valueIterator(name));
682    }
683    copyHeaders(nettyHeaderNames, returnValue, headersInstaller);
684
685    if (needsInputStream(requestObject)) {
686      final EventLoopPinnedByteBufInputStream entityStream =
687        new EventLoopPinnedByteBufInputStream(channelHandlerContext.alloc(),
688                                              channelHandlerContext.executor());
689      assert this.byteBufQueue == null;
690      this.byteBufQueue = entityStream;
691      returnValue.setEntityStream(entityStream);
692    } else {
693      returnValue.setEntityStream(UnreadableInputStream.instance);
694    }
695
696    if (logger.isLoggable(Level.FINER)) {
697      logger.exiting(cn, mn, returnValue);
698    }
699    return returnValue;
700  }

819  private static final boolean needsInputStream(final Object requestObject) {
820    final boolean returnValue;
821    if (requestObject instanceof HttpRequest) {
822      final HttpRequest httpRequest = (HttpRequest)requestObject;
823      returnValue = HttpUtil.getContentLength(httpRequest, -1L) > 0L || HttpUtil.isTransferEncodingChunked(httpRequest);
824    } else {
825      assert requestObject instanceof Http2HeadersFrame;
826      returnValue = !((Http2HeadersFrame)requestObject).isEndStream();
827    }
828    return returnValue;
829  }

831  /**
832   * A utility method that installs a collection of Netty-sourced HTTP
833   * or HTTP/2 headers into the supplied (Jersey) {@link
834   * ContainerRequest}, using the supplied {@link BiConsumer} as a
835   * strategy.
836   *
837   * @param nettyHeaderNames an {@link Iterable} of {@link
838   * CharSequence}-typed header names; may be {@code null} in which
839   * case no action will be taken
840   *
841   * @param target the {@link ContainerRequest} into which the headers
842   * should be installed; must not be {@code null} if {@code
843   * nettyHeaderNames} is non-{@code null}
844   *
845   * @param headersInstaller a {@link BiConsumer} that accepts the
846   * supplied {@link ContainerRequest} and a {@link CharSequence}
847   * representing the current header name whose values should be
848   * installed; must not be {@code null} if {@code nettyHeaderNames}
849   * is non-{@code null}
850   *
851   * @exception NullPointerException if {@code target} or {@code
852   * headersInstaller} is {@code null} and {@code nettyHeaderNames} is
853   * non-{@code null}
854   */
855  public static final void copyHeaders(final Iterable<? extends CharSequence> nettyHeaderNames,
856                                       final ContainerRequest target,
857                                       final BiConsumer<? super ContainerRequest, ? super CharSequence> headersInstaller) {
858    if (nettyHeaderNames != null) {
859      Objects.requireNonNull(target);
860      Objects.requireNonNull(headersInstaller);
861      for (final CharSequence headerName : nettyHeaderNames) {
862        headersInstaller.accept(target, headerName);
863      }
864    }
865  }
