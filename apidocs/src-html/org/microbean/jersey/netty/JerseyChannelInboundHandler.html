168      message instanceof HttpContent ||
169      message instanceof Http2HeadersFrame ||
170      message instanceof Http2DataFrame;

173  /**
174   * Arranges for the {@link ApplicationHandler} {@linkplain
175   * #JerseyChannelInboundHandler(URI, ApplicationHandler)
176   * supplied at construction time} to process the HTTP or HTTP/2
177   * message represented by the supplied {@link Object}.
178   *
179   * <p>This method must be invoked {@linkplain
180   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
181   *
182   * @param channelHandlerContext the {@link ChannelHandlerContext}
183   * representing the current Netty execution; must not be {@code
184   * null}
185   *
186   * @param message the {@link ByteBufHolder}&mdash;an {@link
187   * HttpRequest} or a {@link HttpContent}&mdash;to process; must not
188   * be {@code null}
189   *
190   * @exception Exception if an error occurs
191   *
192   * @see #messageReceived(ChannelHandlerContext, ByteBuf, boolean)
193   */
194  @Override
195  protected final void channelRead0(final ChannelHandlerContext channelHandlerContext, final Object message) throws Exception {
196    Objects.requireNonNull(channelHandlerContext);
197    if (message instanceof HttpRequest || message instanceof Http2HeadersFrame) {
198      this.messageReceived(channelHandlerContext, message);
199    } else if (message instanceof HttpContent || message instanceof Http2DataFrame) {
200      this.messageReceived(channelHandlerContext,
201                           ((ByteBufHolder)message).content(),
202                           message instanceof LastHttpContent || (message instanceof Http2DataFrame) && ((Http2DataFrame)message).isEndStream());
203    } else {
204      assert false; // for now
205    }
206  }

208  private final void messageReceived(final ChannelHandlerContext channelHandlerContext, final Object requestObject) throws Exception {
209    Objects.requireNonNull(channelHandlerContext);
210    assert channelHandlerContext.executor().inEventLoop();
211    assert this.byteBufQueue == null;
212
213    final InjectionManager injectionManager = this.applicationHandler.getInjectionManager();
214    if (injectionManager == null) {
215      throw new IllegalStateException("applicationHandler.getInjectionManager() == null");
216    }
217    
218    final Supplier<? extends ScheduledExecutorService> supplier =
219      () -> injectionManager.getInstance(ScheduledExecutorServiceProvider.class).getExecutorService();
220    final ContainerRequest containerRequest;
221    final ContainerResponseWriter writer;
222    if (requestObject instanceof HttpRequest) {
223      final HttpRequest httpRequest = (HttpRequest)requestObject;
224      containerRequest = this.createContainerRequest(channelHandlerContext, httpRequest);
225      writer = this.createContainerResponseWriter(httpRequest,
226                                                  channelHandlerContext,
227                                                  supplier);
228    } else {
229      assert requestObject instanceof Http2HeadersFrame;
230      final Http2HeadersFrame http2HeadersFrame = (Http2HeadersFrame)requestObject;
231      containerRequest = this.createContainerRequest(channelHandlerContext, http2HeadersFrame);
232      writer = this.createContainerResponseWriter(http2HeadersFrame,
233                                                  channelHandlerContext,
234                                                  supplier);
235    }
236    if (containerRequest == null) {
237      throw new IllegalStateException("createContainerRequest() == null");
238    }
239    if (writer == null) {
240      throw new IllegalStateException("createContainerResponseWriter() == null");
241    }
242    containerRequest.setWriter(writer);
243
244    injectionManager.getInstance(ExecutorServiceProvider.class).getExecutorService().execute(() -> {
245        this.applicationHandler.handle(containerRequest);
246      });
247  }

249  /**
250   * Processes the supplied {@link ByteBuf} representing one of
251   * possibly many "content" portions of an overall HTTP message.
252   *
253   * <p>Internally, the supplied {@code content}, if {@linkplain
254   * ByteBuf#isReadable() readable}, is {@linkplain ByteBuf#retain()
255   * retained} and {@linkplain ByteBufQueue#addByteBuf(ByteBuf) added
256   * to an internal <code>ByteBufQueue</code> implementation} created
257   * by the {@link #channelRead0(ChannelHandlerContext, Object)}
258   * method.
259   *
260   * <p>This method will be invoked {@linkplain
261   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
262   *
263   * @param channelHandlerContext the {@link ChannelHandlerContext}
264   * representing the current Netty execution; must not be {@code
265   * null}
266   *
267   * @param content the {@link ByteBuf} to process; must not be {@code
268   * null}
269   *
270   * @param lastOne {@code true} if {@code content} is known to be the
271   * last such content chunk; {@code false} in all other cases
272   *
273   * @exception Exception if an error occurs
274   *
275   * @see #channelRead0(ChannelHandlerContext, Object)
276   */
277  private final void messageReceived(final ChannelHandlerContext channelHandlerContext, final ByteBuf content, final boolean lastOne) throws Exception {
278    Objects.requireNonNull(channelHandlerContext);
279    Objects.requireNonNull(content);
280    // TODO: in HTTP/2 incoming-payload scenarios this (commented out)
281    // refCnt() returnValue can be 4?!  May have to do with all the
282    // various handlers up front dealing with upgrades etc.
283    //
284    // assert content.refCnt() == 1 : "Unexpected refCnt: " + content.refCnt() + "; thread: " + Thread.currentThread();
285    assert channelHandlerContext.executor().inEventLoop();
286
287    final ByteBufQueue byteBufQueue = this.byteBufQueue;
288
289    if (content.isReadable()) {
290      assert byteBufQueue != null;
291      content.retain();
292      byteBufQueue.addByteBuf(content);
293    }
294
295    if (lastOne && byteBufQueue != null) {
296      this.byteBufQueue = null;
297    }
298  }

370  private final ContainerRequest createContainerRequest(final ChannelHandlerContext channelHandlerContext, final Object requestObject) {
371    Objects.requireNonNull(channelHandlerContext);
372    Objects.requireNonNull(requestObject);
373    assert channelHandlerContext.executor().inEventLoop();
374
375    final String method;
376    final String uriString;
377    final Iterable<? extends CharSequence> nettyHeaderNames;
378   if (requestObject instanceof HttpRequest) {
379      final HttpRequest httpRequest = (HttpRequest)requestObject;
380      final HttpHeaders httpHeaders = httpRequest.headers();
381      assert httpHeaders != null;
382      nettyHeaderNames = httpHeaders.names();
383      method = httpRequest.method().name();
384      uriString = httpRequest.uri();
385    } else {
386      assert requestObject instanceof Http2HeadersFrame;
387      final Http2HeadersFrame http2HeadersFrame = (Http2HeadersFrame)requestObject;
388      final Http2Headers headers = http2HeadersFrame.headers();
389      nettyHeaderNames = headers.names();
390      method = headers.method().toString();
391      uriString = headers.path().toString();
392    }
393    assert method != null;
394    assert uriString != null;
395
396    final SecurityContext securityContext = this.createSecurityContext(channelHandlerContext, requestObject);
397    
398    final ContainerRequest returnValue =
399      new ContainerRequest(this.baseUri,
400                           baseUri.resolve(ContainerUtils.encodeUnsafeCharacters(uriString.startsWith("/") && uriString.length() > 1 ? uriString.substring(1) : uriString)),
401                           method,
402                           securityContext == null ? new SecurityContextAdapter() : securityContext,
403                           new MapBackedPropertiesDelegate());
404
405    final BiConsumer<? super ContainerRequest, ? super CharSequence> headersInstaller;
406    if (requestObject instanceof HttpRequest) {
407      headersInstaller = (containerRequest, name) -> containerRequest.headers(name.toString(), ((HttpRequest)requestObject).headers().getAll(name));
408    } else {
409      assert requestObject instanceof Http2HeadersFrame;
410      headersInstaller = (containerRequest, name) -> containerRequest.headers(name.toString(), (Iterable<CharSequence>)() -> ((Http2HeadersFrame)requestObject).headers().valueIterator(name));
411    }
412    
413    copyHeaders(nettyHeaderNames, returnValue, headersInstaller);
414
415    if (needsInputStream(requestObject)) {
416      final EventLoopPinnedByteBufInputStream entityStream =
417        new EventLoopPinnedByteBufInputStream(channelHandlerContext.alloc(),
418                                              channelHandlerContext.executor());
419      assert this.byteBufQueue == null;
420      this.byteBufQueue = entityStream;
421      returnValue.setEntityStream(entityStream);
422    } else {
423      returnValue.setEntityStream(UnreadableInputStream.instance);
424    }
425
426    return returnValue;
427  }

545  private static final boolean needsInputStream(final Object requestObject) {
546    final boolean returnValue;
547    if (requestObject instanceof HttpRequest) {
548      final HttpRequest httpRequest = (HttpRequest)requestObject;
549      returnValue = HttpUtil.getContentLength(httpRequest, -1L) > 0L || HttpUtil.isTransferEncodingChunked(httpRequest);
550    } else {
551      assert requestObject instanceof Http2HeadersFrame;
552      returnValue = !((Http2HeadersFrame)requestObject).isEndStream();
553    }
554    return returnValue;
555  }

557  /**
558   * A utility method that installs a collection of Netty-sourced HTTP
559   * or HTTP/2 headers into the supplied (Jersey) {@link
560   * ContainerRequest}, using the supplied {@link BiConsumer} as a
561   * strategy.
562   *
563   * @param nettyHeaderNames an {@link Iterable} of {@link
564   * CharSequence}-typed header names; may be {@code null} in which
565   * case no action will be taken
566   *
567   * @param target the {@link ContainerRequest} into which the headers
568   * should be installed; must not be {@code null} if {@code
569   * nettyHeaderNames} is non-{@code null}
570   *
571   * @param headersInstaller a {@link BiConsumer} that accepts the
572   * supplied {@link ContainerRequest} and a {@link CharSequence}
573   * representing the current header name whose values should be
574   * installed; must not be {@code null} if {@code nettyHeaderNames}
575   * is non-{@code null}
576   *
577   * @exception NullPointerException if {@code target} or {@code
578   * headersInstaller} is {@code null} and {@code nettyHeaderNames} is
579   * non-{@code null}
580   */
581  public static final void copyHeaders(final Iterable<? extends CharSequence> nettyHeaderNames,
582                                       final ContainerRequest target,
583                                       final BiConsumer<? super ContainerRequest, ? super CharSequence> headersInstaller) {
584    if (nettyHeaderNames != null) {
585      Objects.requireNonNull(target);
586      Objects.requireNonNull(headersInstaller);
587      for (final CharSequence headerName : nettyHeaderNames) {
588        headersInstaller.accept(target, headerName);
589      }
590    }
591  }
