067/**
068 * A {@link SimpleChannelInboundHandler} that adapts <a
069 * href="https://jersey.github.io/">Jersey</a> to Netty.
070 *
071 * <h2>Thread Safety</h2>
072 *
073 * <p>Instances of this class are safe for concurrent use by multiple
074 * threads, but certain methods must be invoked {@linkplain
075 * EventExecutor#inEventLoop() in the Netty event loop}.</p>
076 *
077 * @author <a href="https://about.me/lairdnelson"
078 * target="_parent">Laird Nelson</a>
079 */

178  /**
179   * Arranges for the {@link ApplicationHandler} {@linkplain
180   * #JerseyChannelInboundHandler(URI, ApplicationHandler, BiFunction)
181   * supplied at construction time} to process the HTTP message
182   * represented by the supplied {@link HttpObject}.
183   *
184   * <p>This method must be invoked {@linkplain
185   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
186   *
187   * @param channelHandlerContext the {@link ChannelHandlerContext}
188   * representing the current Netty execution; must not be {@code
189   * null}
190   *
191   * @param message the {@link HttpObject}&mdash;an {@link
192   * HttpRequest} or a {@link HttpContent}&mdash;to process; must not
193   * be {@code null}
194   *
195   * @exception Exception if an error occurs
196   *
197   * @see #messageReceived(ChannelHandlerContext, HttpRequest)
198   *
199   * @see #messageReceived(ChannelHandlerContext, HttpContent)
200   */
201  @Override
202  protected final void channelRead0(final ChannelHandlerContext channelHandlerContext, final HttpObject message) throws Exception {
203    Objects.requireNonNull(channelHandlerContext);
204    if (message instanceof HttpRequest) {
205      this.messageReceived(channelHandlerContext, (HttpRequest)message);
206    } else if (message instanceof HttpContent) {
207      this.messageReceived(channelHandlerContext, (HttpContent)message);
208    } else {
209      throw new IllegalArgumentException("!(message instanceof HttpRequest || message instanceof HttpContent): " + message);
210    }
211  }

213  /**
214   * Processes the supplied {@link HttpRequest} representing one
215   * portion of an overall HTTP message.
216   *
217   * <p>Internally, a {@link ContainerRequest} is created, a new
218   * {@link NettyContainerResponseWriter} is {@linkplain
219   * ContainerRequest#setWriter(ContainerResponseWriter) installed on
220   * it}, and the {@link ApplicationHandler#handle(ContainerRequest)}
221   * method is invoked on a thread guaranteed not to be the Netty
222   * event loop.</p>
223   *
224   * <p>This method must be invoked {@linkplain
225   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
226   *
227   * @param channelHandlerContext the {@link ChannelHandlerContext}
228   * representing the current Netty execution; must not be {@code
229   * null}
230   *
231   * @param httpRequest the {@link HttpRequest} to process; must not
232   * be {@code null}
233   *
234   * @exception Exception if an error occurs
235   *
236   * @see #createContainerRequest(ChannelHandlerContext, HttpRequest)
237   *
238   * @see ApplicationHandler#handle(ContainerRequest)
239   *
240   * @see ExecutorServiceProvider
241   *
242   * @see NettyContainerResponseWriter
243   */
244  protected void messageReceived(final ChannelHandlerContext channelHandlerContext, final HttpRequest httpRequest) throws Exception {
245    Objects.requireNonNull(channelHandlerContext);
246    assert channelHandlerContext.executor().inEventLoop();
247
248    assert this.byteBufQueue == null;
249
250    final InjectionManager injectionManager = this.applicationHandler.getInjectionManager();
251    if (injectionManager == null) {
252      throw new IllegalStateException("applicationHandler.getInjectionManager() == null");
253    }
254
255    final ContainerResponseWriter writer =
256      this.createNettyContainerResponseWriter(httpRequest,
257                                              channelHandlerContext,
258                                              () -> injectionManager.getInstance(ScheduledExecutorServiceProvider.class).getExecutorService());
259    if (writer == null) {
260      throw new IllegalStateException("createNettyContainerResponseWriter() == null");
261    }
262
263    final ContainerRequest containerRequest = this.createContainerRequest(channelHandlerContext, httpRequest);
264    if (containerRequest == null) {
265      throw new IllegalStateException("createContainerRequest() == null");
266    }
267    containerRequest.setWriter(writer);
268
269    injectionManager.getInstance(ExecutorServiceProvider.class).getExecutorService().execute(() -> {
270        this.applicationHandler.handle(containerRequest);
271      });
272  }

277   *
278   * <p>This method never returns {@code null}.</p>

280   * <p>Overrides of this method must not return {@code null}.</p>

282   * <p>This implementation simply invokes the {@link

286   *
287   * <p>In normal usage, this method is invoked {@linkplain
288   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
289   *

291   * #messageReceived(ChannelHandlerContext, HttpRequest)} method.

293   * result.</p>
294   *
295   * @param httpRequest the {@link HttpRequest} being processed; must
296   * not be {@code null}
297   *
298   * @param channelHandlerContext the {@link ChannelHandlerContext}
299   * representing the current Netty execution; must not be {@code
300   * null}
301   *
302   * @param scheduledExecutorServiceSupplier a {@link Supplier} that
303   * can {@linkplain Supplier#get() supply} a {@link

305   *
306   * @return a new {@link NettyContainerResponseWriter}; never {@code
307   * null}
308   *
309   * @see
310   * NettyContainerResponseWriter#NettyContainerResponseWriter(HttpRequest,

312   *
313   * @see #messageReceived(ChannelHandlerContext, HttpRequest)
314   */
315  protected NettyContainerResponseWriter createNettyContainerResponseWriter(final HttpRequest httpRequest,

321  /**
322   * Processes the supplied {@link HttpContent} representing one
323   * portion of an overall HTTP message.
324   *
325   * <p>Internally, the {@link HttpContent}'s {@linkplain
326   * HttpContent#content() content}, if {@linkplain
327   * ByteBuf#isReadable() readable}, is {@linkplain ByteBuf#retain()
328   * retained} and {@linkplain ByteBufQueue#addByteBuf(ByteBuf) added
329   * to an internal <code>ByteBufQueue</code> implementation} created
330   * by the {@link #messageReceived(ChannelHandlerContext,
331   * HttpRequest)} method.  Then, if the {@link HttpContent} is an
332   * instance of {@link LastHttpContent}, that internal {@link
333   * ByteBufQueue} is {@linkplain ByteBufQueue#close() closed}.</p>
334   *
335   * <p>This method must be invoked {@linkplain
336   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
337   *
338   * @param channelHandlerContext the {@link ChannelHandlerContext}
339   * representing the current Netty execution; must not be {@code
340   * null}
341   *
342   * @param httpContent the {@link HttpContent} to process; must not
343   * be {@code null}
344   *
345   * @exception Exception if an error occurs
346   */
347  protected void messageReceived(final ChannelHandlerContext channelHandlerContext, final HttpContent httpContent) throws Exception {
348    Objects.requireNonNull(channelHandlerContext);
349    Objects.requireNonNull(httpContent);
350    assert channelHandlerContext.executor().inEventLoop();
351
352    final ByteBufQueue byteBufQueue = this.byteBufQueue;
353    
354    final ByteBuf content = httpContent.content();
355    assert content != null;
356    assert content.refCnt() == 1 : "Unexpected refCnt: " + content.refCnt() + "; thread: " + Thread.currentThread();
357
358    if (content.isReadable()) {
359      content.retain();
360      assert byteBufQueue != null;
361      byteBufQueue.addByteBuf(content);
362    }
363
364    if (byteBufQueue != null && httpContent instanceof LastHttpContent) {
365      try {
366        this.byteBufQueue.close();
367      } finally {
368        this.byteBufQueue = null;
369      }
370    }
371  }

373  /**
374   * Creates a {@link ContainerRequest} representing the supplied
375   * {@link HttpRequest} and returns it.
376   *
377   * <p>This method never returns {@code null}.</p>
378   *
379   * <p>Internally, this method sets up an internal {@link
380   * ByteBufQueue} if necessary representing incoming entity data.
381   * The {@link ByteBufQueue} in question is an instance of {@link
382   * EventLoopPinnedByteBufInputStream} and is also {@linkplain
383   * ContainerRequest#setEntityStream(InputStream) installed} on the
384   * {@link ContainerRequest}.</p>
385   *
386   * <p>The internals of this method also {@linkplain
387   * ByteBufAllocator#compositeBuffer() may allocate a
388   * <code>CompositeByteBuf</code>} which will be {@linkplain
389   * ByteBuf#release() released} when the {@linkplain
390   * Channel#closeFuture() underlying <code>Channel</code>
391   * closes}.</p>
392   *
393   * <p>This method must be invoked {@linkplain
394   * EventExecutor#inEventLoop() in the Netty event loop}.</p>
395   *
396   * @param channelHandlerContext the {@link ChannelHandlerContext}
397   * representing the current Netty execution; must not be {@code
398   * null}
399   *
400   * @param httpRequest the {@link HttpRequest} to process; must not
401   * be {@code null}
402   *
403   * @return a new {@link ContainerRequest}; never {@code null}
404   *
405   * @see ContainerRequest
406   *
407   * @see EventLoopPinnedByteBufInputStream
408   *
409   * @see ByteBufQueue
410   */
411  protected ContainerRequest createContainerRequest(final ChannelHandlerContext channelHandlerContext, final HttpRequest httpRequest) {
412    Objects.requireNonNull(channelHandlerContext);
413    Objects.requireNonNull(httpRequest);
414    assert channelHandlerContext.executor().inEventLoop();
415
416    final String uriString = httpRequest.uri();
417    assert uriString != null;
418
419    final SecurityContext securityContext = this.createSecurityContext(channelHandlerContext, httpRequest);
420    final ContainerRequest returnValue =
421      new ContainerRequest(this.baseUri,
422                           baseUri.resolve(ContainerUtils.encodeUnsafeCharacters(uriString.startsWith("/") && uriString.length() > 1 ? uriString.substring(1) : uriString)),
423                           httpRequest.method().name(),
424                           securityContext == null ? new SecurityContextAdapter() : securityContext,
425                           new MapBackedPropertiesDelegate());
426
427    final HttpHeaders headers = httpRequest.headers();
428    if (headers != null) {
429      final Iterable<? extends String> headerNames = headers.names();
430      if (headerNames != null) {
431        for (final String headerName : headerNames) {
432          if (headerName != null) {
433            returnValue.headers(headerName, headers.getAll(headerName));
434          }
435        }
436      }
437    }
438
439    if (HttpUtil.getContentLength(httpRequest, -1L) > 0L || HttpUtil.isTransferEncodingChunked(httpRequest)) {
440
441      final CompositeByteBuf compositeByteBuf = channelHandlerContext.alloc().compositeBuffer();
442      assert compositeByteBuf != null;
443      channelHandlerContext.channel().closeFuture().addListener(ignored -> compositeByteBuf.release());
444
445      final EventLoopPinnedByteBufInputStream entityStream = new EventLoopPinnedByteBufInputStream(compositeByteBuf, channelHandlerContext.executor());
446      assert this.byteBufQueue == null;
447      this.byteBufQueue = entityStream;
448      returnValue.setEntityStream(entityStream);
449
450    } else {
451      returnValue.setEntityStream(UnreadableInputStream.instance);
452    }
453
454    return returnValue;
455  }

457  /**
458   * Returns a new {@link SecurityContext} instance on each invocation.
459   *
460   * @param channelHandlerContext the {@link ChannelHandlerContext}
461   * representing the current Netty execution; supplied for
462   * convenience; must not be {@code null}
463   *
464   * @param httpRequest the current {@link HttpRequest}; supplied for
465   * convenience; must not be {@code null}
466   *
467   * @return a new {@link SecurityContext} instance on each invocation
468   *
469   * @see SecurityContextAdapter
470   */
471  private final SecurityContext createSecurityContext(final ChannelHandlerContext channelHandlerContext, final HttpRequest httpRequest) {
472    if (this.securityContextBiFunction != null) {
473      return this.securityContextBiFunction.apply(channelHandlerContext, httpRequest);
474    } else {
475      return new SecurityContextAdapter();
476    }
477  }
