051/**
052 * An {@link AbstractNettyContainerResponseWriter} that works with
053 * {@link HttpRequest}-typed request objects.
054 *
055 * @author <a href="https://about.me/lairdnelson"
056 * target="_parent">Laird Nelson</a>
057 *
058 * @see AbstractNettyContainerResponseWriter
059 *
060 * @see Http2ContainerResponseWriter
061 */
062public class HttpContainerResponseWriter extends AbstractNettyContainerResponseWriter<HttpRequest> {
063
064
065
066  /*
067   * Static fields.
068   */
069
070
071  private static final String cn = HttpContainerResponseWriter.class.getName();
072
073  private static final Logger logger = Logger.getLogger(cn);
074
075
076  /*
077   * Constructors.
078   */
079
080
081  /**
082   * Creates a new {@link HttpContainerResponseWriter}.
083   *
084   * @param httpRequest the {@link HttpRequest} being responded to;
085   * must not be {@code null}
086   *
087   * @param channelHandlerContext the {@link ChannelHandlerContext}
088   * representing the current Netty execution; must not be {@code
089   * null}
090   *
091   * @param scheduledExecutorServiceSupplier a {@link Supplier} that
092   * can {@linkplain Supplier#get() supply} a {@link
093   * ScheduledExecutorService}; must not be {@code null}
094   *
095   * @exception NullPointerException if any of the parameters is
096   * {@code null}
097   */
098  public HttpContainerResponseWriter(final HttpRequest httpRequest,
099                                     final ChannelHandlerContext channelHandlerContext,
100                                     final Supplier<? extends ScheduledExecutorService> scheduledExecutorServiceSupplier) {
101    super(httpRequest, channelHandlerContext, scheduledExecutorServiceSupplier);
102  }
103
104
105  /*
106   * Instance methods.
107   */
108
109
110  /**
111   * Implements the {@link
112   * AbstractNettyContainerResponseWriter#writeAndFlushStatusAndHeaders(ContainerResponse,
113   * long, ChannelPromise)} method by {@linkplain
114   * ChannelHandlerContext#writeAndFlush(Object) writing and flushing}
115   * an {@link HttpResponse} object containing a relevant {@link
116   * HttpResponseStatus} object.
117   *
118   * @param containerResponse the {@link ContainerResponse} being
119   * processed; must not be {@code null}
120   *
121   * @param contentLength the length of the content in bytes; will be
122   * less than {@code 0} if the content length is unknown
123   *
124   * @param channelPromise a {@link ChannelPromise} to pass to any
125   * write operation; must not be {@code null}
126   *
127   * @exception NullPointerException if {@code containerResponse} or
128   * {@code channelPromise} is {@code null}
129   *
130   * @see #writeResponseStatusAndHeaders(long, ContainerResponse)
131   */
132  @Override
133  protected final void writeAndFlushStatusAndHeaders(final ContainerResponse containerResponse,
134                                                     final long contentLength,
135                                                     final ChannelPromise channelPromise) {
136    final String mn = "writeAndFlushStatusAndHeaders";
137    if (logger.isLoggable(Level.FINER)) {
138      logger.entering(cn, mn, new Object[] { containerResponse, Long.valueOf(contentLength) });
139    }
140    Objects.requireNonNull(containerResponse);
141    Objects.requireNonNull(channelPromise);
142
143    final String reasonPhrase = containerResponse.getStatusInfo().getReasonPhrase();
144    final HttpResponseStatus status = reasonPhrase == null ? HttpResponseStatus.valueOf(containerResponse.getStatus()) : new HttpResponseStatus(containerResponse.getStatus(), reasonPhrase);
145
146    final HttpResponse httpResponse;
147    if (contentLength < 0L) {
148      httpResponse = new DefaultHttpResponse(this.requestObject.protocolVersion(), status);
149      HttpUtil.setTransferEncodingChunked(httpResponse, true);
150    } else if (contentLength == 0L) {
151      httpResponse = new DefaultFullHttpResponse(this.requestObject.protocolVersion(), status);
152      HttpUtil.setContentLength(httpResponse, 0L);
153    } else {
154      httpResponse = new DefaultHttpResponse(this.requestObject.protocolVersion(), status);
155      HttpUtil.setContentLength(httpResponse, contentLength);
156    }
157
158    final HttpHeaders nettyHeaders = httpResponse.headers();
159    assert nettyHeaders != null;
160    copyHeaders(containerResponse.getStringHeaders(), UnaryOperator.identity(), nettyHeaders::add);
161    if (HttpUtil.isKeepAlive(this.requestObject)) {
162      HttpUtil.setKeepAlive(httpResponse, true);
163    }
164    this.channelHandlerContext.writeAndFlush(httpResponse, channelPromise);
165
166    if (logger.isLoggable(Level.FINER)) {
167      logger.exiting(cn, mn);
168    }
169  }
170
171  /**
172   * Implements the {@link
173   * AbstractNettyContainerResponseWriter#needsOutputStream(long)}
174   * method by returning {@code true} if and only if the supplied
175   * {@code contentLength} is not equal to {@code 0L} and if the
176   * {@linkplain HttpRequest#method() request method} is not {@link
177   * HttpMethod#HEAD HEAD}.
178   *
179   * @param contentLength the length of the content in bytes; will be
180   * less than {@code 0} if the content length is unknown
181   *
182   * @return {@code true} if {@code contentLength} is not equal to
183   * {@code 0L} and if the {@linkplain HttpRequest#method() request
184   * method} is not {@link HttpMethod#HEAD HEAD}; {@code false} in all
185   * other cases
186   */
187  @Override
188  protected final boolean needsOutputStream(final long contentLength) {
189    return contentLength != 0L && !HttpMethod.HEAD.equals(this.requestObject.method());
190  }
191
192  /**
193   * Returns a new {@link FunctionalByteBufChunkedInput
194   * FunctionalByteBufChunkedInput&lt;ByteBuf&gt;} when invoked.
195   *
196   * @param eventExecutor {@inheritDoc}
197   *
198   * @param source {@inheritDoc}
199   *
200   * @param contentLength {@inheritDoc}
201   *
202   * @return a new {@link ByteBufChunkedInput}; never {@code null}
203   *
204   * @exception NullPointerException if {@code eventExecutor} or
205   * {@code source} is {@code null}
206   *
207   * @see AbstractNettyContainerResponseWriter#createChunkedInput(EventExecutor, ByteBuf, long)
208   *
209   * @see ByteBufChunkedInput#ByteBufChunkedInput(ByteBuf, long)
210   *
211   * @see ChunkedInput#readChunk(ByteBufAllocator)
212   */
213  @Override
214  protected final BoundedChunkedInput<?> createChunkedInput(final EventExecutor eventExecutor, final ByteBuf source, final long contentLength) {
215    final String mn = "createChunkedInput";
216    if (logger.isLoggable(Level.FINER)) {
217      logger.entering(cn, mn, new Object[] { eventExecutor, source, Long.valueOf(contentLength) });
218    }
219    final BoundedChunkedInput<?> returnValue = new FunctionalByteBufChunkedInput<ByteBuf>(source, UnaryOperator.identity(), contentLength);
220    if (logger.isLoggable(Level.FINER)) {
221      logger.exiting(cn, mn, returnValue);
222    }
223    return returnValue;
224  }
225
226  /**
227   * {@linkplain ChannelHandlerContext#write(Object) Writes} {@link
228   * LastHttpContent#EMPTY_LAST_CONTENT} when invoked.
229   *
230   * @param channelPromise a {@link ChannelPromise} to pass to any
231   * write operation; must not be {@code null}
232   *
233   * @exception NullPointerException if {@code channelPromise} is
234   * {@code null}
235   *
236   * @see LastHttpContent
237   *
238   * @see AbstractNettyContainerResponseWriter#writeLastContentMessage(ChannelPromise)
239   */
240  @Override
241  protected final void writeLastContentMessage(final ChannelPromise channelPromise) {
242    Objects.requireNonNull(channelPromise);
243    // Send the magic message that tells the HTTP machinery to
244    // finish up.
245    this.channelHandlerContext.write(LastHttpContent.EMPTY_LAST_CONTENT, channelPromise);
246  }
247
248  /**
249   * {@linkplain ChannelHandlerContext#write(Object) Writes} a new
250   * {@link DefaultFullHttpResponse} message with a {@code
251   * Content-Length} of {@code 0} and a status equal to {@link
252   * HttpResponseStatus#INTERNAL_SERVER_ERROR}.
253   *
254   * @param channelPromise a {@link ChannelPromise} to pass to any
255   * write operation; must not be {@code null}
256   *
257   * @exception NullPointerException if {@code channelPromise} is
258   * {@code null}
259   *
260   * @see
261   * AbstractNettyContainerResponseWriter#writeFailureMessage(ChannelPromise)
262   */
263  @Override
264  protected final void writeFailureMessage(final ChannelPromise channelPromise) {
265    Objects.requireNonNull(channelPromise);
266    final HttpMessage failureMessage = new DefaultFullHttpResponse(this.requestObject.protocolVersion(),
267                                                                   HttpResponseStatus.INTERNAL_SERVER_ERROR);
268    HttpUtil.setContentLength(failureMessage, 0L);
269    this.channelHandlerContext.write(failureMessage, channelPromise);
270  }
