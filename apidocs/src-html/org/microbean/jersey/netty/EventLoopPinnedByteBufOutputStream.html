017package org.microbean.jersey.netty;
018
019import java.io.IOException;
020import java.io.OutputStream;
021
022import java.util.Objects;
023
024import io.netty.buffer.ByteBuf;
025
026import io.netty.util.concurrent.EventExecutor;
027import io.netty.util.concurrent.Future;
028import io.netty.util.concurrent.GenericFutureListener;
029
030/**
031 * An {@link OutputStream} whose operations occur on Netty's event
032 * loop.
033 *
034 * @author <a href="https://about.me/lairdnelson"
035 * target="_parent">Laird Nelson</a>
036 */
037public final class EventLoopPinnedByteBufOutputStream extends OutputStream {
038
039
040  /*
041   * Instance fields.
042   */
043
044  
045  private final EventExecutor eventExecutor;
046
047  private final ByteBuf byteBuf;
048
049  private final GenericFutureListener<? extends Future<? super Void>> listener;
050  
051  /*
052   * Constructors.
053   */
054
055  
056  /**
057   * Creates a new {@link EventLoopPinnedByteBufOutputStream}.
058   *
059   * @param eventExecutor an {@link EventExecutor} that will be used
060   * to {@linkplain EventExecutor#inEventLoop() ensure all operations
061   * are executed on the Netty event loop}; must not be {@code null}
062   *
063   * @param byteBuf the {@link ByteBuf} that will be {@linkplain
064   * ByteBuf#writeBytes(byte[], int, int) written to}; must not be
065   * {@code null}
066   *
067   * @param listener a {@link GenericFutureListener} that {@linkplain
068   * GenericFutureListener#operationComplete(Future) will be notified
069   * when a <code>Future</code> representing a <code>ByteBuf</code>
070   * write operation completes on the Netty event loop}; may be {@code
071   * null}
072   */
073  public EventLoopPinnedByteBufOutputStream(final EventExecutor eventExecutor,
074                                            final ByteBuf byteBuf,
075                                            final GenericFutureListener<? extends Future<? super Void>> listener) {
076    super();
077    this.eventExecutor = Objects.requireNonNull(eventExecutor);
078    this.byteBuf = Objects.requireNonNull(byteBuf);
079    this.listener = listener;
080  }
081
082
083  /*
084   * Instance methods.
085   */
086
087  
088  @Override
089  public final void write(final byte[] bytes) throws IOException {
090    this.perform(bb -> bb.writeBytes(bytes));
091  }
092
093  @Override
094  public final void write(final byte[] bytes, final int offset, final int length) throws IOException {
095    this.perform(bb -> bb.writeBytes(bytes, offset, length));
096  }
097
098  @Override
099  public final void write(final int b) throws IOException {
100    this.perform(bb -> bb.writeByte(b));
101  }
102
103  private final void perform(final ByteBufOperation byteBufOperation) throws IOException {
104    if (this.eventExecutor.inEventLoop()) {
105      byteBufOperation.applyTo(this.byteBuf);
106    } else {
107      final Future<Void> byteBufOperationFuture = this.eventExecutor.submit(() -> {
108          byteBufOperation.applyTo(this.byteBuf);
109          return null;
110        });
111      assert byteBufOperationFuture != null;
112      if (this.listener != null) {
113        byteBufOperationFuture.addListener(this.listener);
114      }
115    }
116  }
117
118
119  /*
120   * Inner and nested classes.
121   */
122  
123  
124  /**
125   * A {@linkplain FunctionalInterface functional interface} whose
126   * implementations typically read from or write to a given {@link
127   * ByteBuf}.
128   *
129   * @author <a href="https://about.me/lairdnelson"
130   * target="_parent">Laird Nelson</a>
131   *
132   * @see #applyTo(ByteBuf)
133   */
134  @FunctionalInterface
135  private static interface ByteBufOperation {
136
137    /**
138     * Operates on the supplied {@link ByteBuf} in some way.
139     *
140     * @param target the {@link ByteBuf} to operate on; must not be
141     * {@code null}
142     *
143     * @exception IOException if an error occurs
144     */
145    public void applyTo(final ByteBuf target) throws IOException;
146    
147  }
148
149}
